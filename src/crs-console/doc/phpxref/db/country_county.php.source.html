  17   * Returns array of data for counties and countries, getting data and filling cache if necessary
  18   * 
  19   * @return array county and country data (see {@link _get_country_county})
  20   */
  21  function get_country_county()
  22      {
  23      $timeout = 86400; // max-age = 1 day
  24      $cache_key = 'data_get_country_county';
  25      $s = Yii::app()->cache->get($cache_key);
  26      if($s==null)
  27          {
  28          $s = _get_country_county();
  29          Yii::app()->cache->set($cache_key, $s, $timeout);
  30          }
  31      return $s;
  32      }

  34  /**
  35   * Returns array of data for counties and countries
  36   * 
  37   * Returned array contains three sub-arrays;
  38   * 
  39   * - 'country_names' : maps country-name to country-code
  40   * 
  41   * - 'country_codes' : maps country code to country name
  42   * 
  43   * - 'countrycode_counties' : maps country-code to county
  44   * 
  45   * @return array county and country data
  46   */
  47  function _get_country_county()
  48      {
  49      $data = array();
  50      $data['country_names'] = array();
  51      $data['country_codes'] = array();
  52      $data['countrycode_counties'] = array();
  53  
  54      $d = DB::get_db();
  55      $sql = 'SELECT c.CountryCode, c.Country, County FROM Countries c LEFT JOIN Counties ON Counties.CountryCode = c.CountryCode order by c.Country, County;';
  56      $d->query($sql,__FILE__,__LINE__);
  57  
  58      while($r = $d->fetch())
  59          {
  60          $data['country_names'][$r->Country] = $r->CountryCode;
  61          $data['country_codes'][$r->CountryCode] = $r->Country;
  62          // breaks selection, model/api needs string instead of code ...
  63          //$data['countrycode_counties'][$r->CountryCode][] = str_replace(array('Co. ','Co ','CO. ','CO '),'',$r->County);
  64          $data['countrycode_counties'][$r->CountryCode][] = $r->County;
  65          }
  66      //Yii::log('returning '.print_r($data,true), 'debug', 'get_country_county()');
  67      return $data;
  68      }

  86   * get County options in a format suitable for a Yii $form->dropDownList
  87   * 
  88   * @param  string $country country name
  89   * @return array   select-list options array
  90   */
  91  function getCountyOptions($country)
  92      {
  93      $country_county_data = get_country_county();
  94      $counties = array();
  95      // find country code
  96      $ccode = $country_county_data['country_names'][$country];
  97      if($ccode != null and is_array($country_county_data['countrycode_counties'][$ccode]))
  98          foreach($country_county_data['countrycode_counties'][$ccode] as $county)
  99              if($county!=null and strlen($county)>1)
 100                  $counties[$county] = $county;
 101      if(count($counties)==0)
 102          $counties['N/A'] = 'N/A';
 103      //Yii::log('returning '.print_r($counties,true), 'debug', "getCountyOptions($country)");
 104      return $counties;
 105      }
