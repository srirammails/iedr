  17   * get current time as a floating-point number including microseconds
  18   * 
  19   * @return float time
  20   */
  21  function microtime_float()
  22      {
  23      list($usec, $sec) = explode(' ', microtime());
  24      return ((float)$usec + (float)$sec);
  25      }

  27  /**
  28   * runs passed string as a sql query and returns first row's column 'x', or null
  29   * 
  30   * @param  string $sql sql statement
  31   * @return mixed  column 'x' of first row of results

  33  function run_stats_qry($sql)
  34      {
  35      $d=DB::get_db();
  36      $d->query($sql,__FILE__,__LINE__);
  37      if($r=$d->fetch())
  38          return $r->x;
  39      return null;

  42  /**
  43   * get cached value with passed key, or, if not cached, run passed query and return column 'x' of first row (and add to cache)
  44   * 
  45   * @param  string $cache_key cache key
  46   * @param  string $sql sql statement
  47   * @return mixed column 'x' of first row of query results
  48   */
  49  function get_cached_stats_query($cache_key,$sql)
  50      {
  51      // check cache
  52      $s = Yii::app()->cache->get($cache_key);
  53      if($s==null)
  54          {
  55          $tstart = microtime_float();
  56          $s = run_stats_qry($sql);
  57          $tend = microtime_float();
  58          $elapsed = $tend - $tstart;
  59          $timeout = 600; // 10 min
  60          Yii::app()->cache->set($cache_key, $s, $timeout);
  61          #Yii::log('query for '.$cache_key.' took '.$elapsed.' seconds', 'debug', 'short_account_stats::get_cached_stats_query()'.__LINE__);
  62          }
  63      return $s;
  64      }

 197              else
 198                  Yii::log(print_r($errs,true), 'error', 'short_account_stats::get_stats_DepositBalance()'.__LINE__);
 199              if(!strpos(print_r($errs,true),'DepositNotFoundException')===false)
 200                  {
 201                  Yii::log('DepositNotFoundException : setting deposit to 0', 'error', 'short_account_stats::get_stats_DepositBalance()'.__LINE__);
 202                  $result = new CRSPaymentAppService_depositVO();
 203                  $result->closeBal = 0;
 204                  }
 205              $timeout = 60; // secs
 206              Yii::app()->cache->set($cache_key, $result, $timeout);
 207              }
 208          catch(Exception $e)

 210              $errstr = print_r($errs,true);

 214      #else
