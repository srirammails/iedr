  67  /**
  68   * converts a CRS-WS-API Tech Status string to Version for use in Web Page Display
  69   * 
  70   * @param  string $wsapi_str CRS-WS-API Ticket Tech Status
  71   * @return string Web-page version of Tech status string
  72   */
  73  function decipher_tech_status($wsapi_str)
  74      {
  75      global $tech_sts_wsapi_strings, $tech_sts_console_strings;
  76      $n = 0;
  77      foreach($tech_sts_wsapi_strings as $k => $v)
  78          if($v == $wsapi_str)
  79              { $n = $k; break; }
  80      $r = isset($tech_sts_console_strings[$n]) ? $tech_sts_console_strings[$n] : 'Undefined';
  81      Yii::log('decipher_tech_status('.$wsapi_str.')=>'.$r, 'info', '');
  82      return $r;
  83      }
