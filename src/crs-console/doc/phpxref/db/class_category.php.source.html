  32   * @return array as returned by {@link _get_full_class_category}

  34  function get_full_class_category()
  35      {
  36      $timeout = 300;
  37      $cache_key = 'data_get_full_class_category';
  38      $s = Yii::app()->cache->get($cache_key);
  39      if($s==null)
  40          {
  41          $s = _get_full_class_category();
  42          Yii::app()->cache->set($cache_key, $s, $timeout);
  43          }
  44      return $s;
  45      }

  47  /**
  48   * gets class and category arrays
  49   * 
  50   * returned array contains three subarrays:
  51   * 
  52   * - 'classcat' : keys are Class names, values are arrays with key strings like 'Class:Category' and values of category strings
  53   * 
  54   * - 'class' : keys are class-ID, values are class names
  55   * 
  56   * - 'category' : keys are category-ID, values are category names
  57   * 
  58   * 
  59   * @return array (for array structure, see above)
  60   */
  61  function _get_full_class_category()
  62      {
  63      $d = DB::get_db();
  64      $sql = 'SELECT tc.name as cls, ctc.class_id, tt.name as cat, tt.id as cat_id '.
  65              'FROM Class tc, Category tt, Class_to_Category ctc '.
  66              'WHERE tt.id = ctc.category_id '.
  67              'AND tc.id = ctc.class_id '.
  68              'order by 1,3;';
  69      $d->query($sql,__FILE__,__LINE__);
  70      $class_category = array();
  71      while($r = $d->fetch())
  72          {
  73          $class_category['classcat'][$r->cls][$r->cls .':'. $r->cat] = $r->cat;
  74          $class_category['class'][$r->class_id] = $r->cls;
  75          $class_category['category'][$r->cat_id] = $r->cat;
  76          }
  77      return $class_category;
  78      }

  80  /**
  81   * returns a class-name string as found in the database, matching the passed parameter case-insensitively
  82   * 
  83   * @param  string $c class name

  86  function encode_class_to_wsapi_str($c)
  87      {
  88      // convert console display /class/ to XML-API string
  89      $s = null;
  90      $d = get_full_class_category();
  91      $dd = $d['class'];
  92      if(is_array($dd))
  93          {
  94          foreach($dd as $id => $cls)
  95              if(preg_match('/.*'.$c.'.*/i',$cls)!==0)
  96                  { $s = $dd[$id]; break; }
  97          Yii::log('encode_class_to_wsapi_str('.$c.')=>'.$s, 'info', '');
  98          }
  99      else
 100          {
 101          Yii::log('encode_class_to_wsapi_str() no array in'.print_r($d,true), 'ERROR', '');
 102          }
 103      return $s;
 104      }
