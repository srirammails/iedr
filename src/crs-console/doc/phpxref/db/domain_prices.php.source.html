  17  /**
  18   * returns cached domain pricing info array, getting data and filling cache if necessary
  19   * 
  20   * @return array pricing data, see {@link _get_reg_prices}
  21   */
  22  function get_reg_prices()
  23      {
  24      $timeout = 9000;
  25      $cache_key = 'data_get_domain_prices';
  26      $s = Yii::app()->cache->get($cache_key);
  27      if($s==null)
  28          {
  29          $s = _get_reg_prices();
  30          Yii::app()->cache->set($cache_key, $s, $timeout);
  31          }
  32      return $s;
  33      }

  35  /**
  36   * Returns domain pricing information
  37   * 
  38   * Returns array with four sub-arrays;
  39   * 
  40   * - 'CODE' : maps all product IDs to product description
  41   * 
  42   * - 'REG' : maps only registration product IDs to product descriptions
  43   * 
  44   * - 'REN' : maps only renewal product IDs to product descriptions
  45   * 
  46   * - 'DEFAULT' : maps string keys 'REN' and 'REG' to default renewal and registration product IDs respectively
  47   * 
  48   * @return array domain pricing data
  49   */
  50  function _get_reg_prices()
  51      {
  52      $regprice = array();
  53      // get data with code
  54      $regprice['CODE'] = array();
  55      // get Registration list
  56      $regprice['REG'] = array();
  57      // get Renewal list
  58      $regprice['REN'] = array();
  59      // defaults
  60      $regprice['DEFAULT'] = array();
  61  
  62      $result = null;
  63      $errs = null;
  64      CRSDomainAppService_service::getDomainPricing($result,$errs, Yii::app()->user->authenticatedUser);
  65      if($errs==null and is_array($result))
  66          {
  67          foreach($result as $p)
  68              {
  69              $regprice['CODE'][$p->id] = $p;
  70              if($p->forRegistration)
  71                  {
  72                  $regprice['REG'][$p->id] = $p;
  73                  if($p->defaultPrice)
  74                      $regprice['DEFAULT']['REG'] = $p->id;
  75                  }
  76              if($p->forRenewal)
  77                  {
  78                  $regprice['REN'][$p->id] = $p;
  79                  if($p->defaultPrice)
  80                      $regprice['DEFAULT']['REN'] = $p->id;
  81                  }
  82              }
  83          }
  84      return $regprice;
  85      }
