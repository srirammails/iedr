 262      /**
 263       * attempts to create a mysql connection resource
 264       * 
 265       * connects and sets database and various parameters:
 266       * 
 267       * - character set 'utf8'
 268       * 
 269       * - isolation level 'read committed'
 270       * 
 271       * Will set {@last_errno} for connection attempt
 272       * 
 273       * 
 274       * @param  string   $host hostname or IP address
 275       * @param  string   $user username
 276       * @param  string   $pass password
 277       * @param  string   $db   database
 278       * @return object   mysql connection resource
 279       * @access protected
 280       */
 281  	protected function connect($host, $user, $pass, $db)
 282          {
 283          $connection = @mysqli_connect($host, $user, $pass, $db);
 284          $this->last_errno = @mysqli_connect_error();
 285          if($this->last_errno==0 and $connection!=null and $connection!=false)
 286              {
 287              @mysqli_set_charset($connection, 'utf8');
 288              $this->last_errno = @mysqli_errno($connection);
 289              if($this->last_errno==0)
 290                  {
 291                  $this->curr_qry = @mysqli_query($connection,"SET SESSION sql_mode='STRICT_TRANS_TABLES,STRICT_ALL_TABLES,NO_ZERO_IN_DATE,NO_ZERO_DATE,ERROR_FOR_DIVISION_BY_ZERO,NO_AUTO_CREATE_USER';");
 292                  $this->last_errno = @mysqli_errno($connection);
 293                  if($this->last_errno==0)
 294                      {
 295                      $this->curr_qry = @mysqli_query($connection,"SET SESSION TRANSACTION ISOLATION LEVEL READ COMMITTED;");
 296                      $this->last_errno = @mysqli_errno($connection);
 297                      if($this->last_errno!=0)
 298                          $this->error('set isolation level',__FILE__,__LINE__);
 299                      }
 300                  else
 301                      $this->error('set sql mode',__FILE__,__LINE__);
 302                  }
 303              else
 304                  $this->error('set charset utf8',__FILE__,__LINE__);
 305              }
 306          else
 307              $this->error('connect',__FILE__,__LINE__);
 308          return $connection;
 309          }

 422      /**
 423       * performs the supplied sql statement, selecting the appropriate connection and handling results as needed
 424       * 
 425       * uses read-connection for selects, and for inserts, tries to get the new-row-ID
 426       * 
 427       * @param  string  $typ        Parameter description (if any) ...
 428       * @param  string  $sql        sql statement
 429       * @param  string  $f          originating php filename, for logging
 430       * @param  string  $l          originating php file linenumber, for logging
 431       * @param  boolean $autocommit true to set autocommit on
 432       * @return string  null on connection failure, boolean outcome of statement, or integer on successful insert into a table with an auto-increment integer primary key
 433       * @access protected
 434       */
 435  	protected function perform($typ,$sql,$f,$l,$autocommit=true)
 436          {
 437          $this->curr_qry = null;
 438          $this->last_sql =& $sql;
 439          $this->last_sql_f = $f;
 440          $this->last_sql_l = $l;
 441          $this->last_errno = null;
 442          $conn = $typ=='s' ? $this->get_read_connection() : $this->get_write_connection($autocommit);
 443          if($conn)
 444              {
 445              $this->curr_qry = @mysqli_query($conn,$sql);
 446              $this->savestate();
 447              if($this->last_errno==0)
 448                  {
 449                  return ($typ=='i') ? @mysqli_insert_id($conn) : $this->curr_qry;
 450                  }
 451              }
 452          else
 453              $this->error('query: null connection');
 454          $this->curr_qry = null; // otherwise 'false'
 455          return null;
 456          }
