 187  	public function getPartialQuery($model,$criteria)
 188          {
 189          $u = Yii::app()->user->name;
 190          $dt = '';
 191          switch($this->type)
 192              {
 193              case 'XferTo':
 194                  $dt = $this->getDateStr(array('Moved_Dt'),$model); break;
 195              case 'nrc_adv_pay_inv_hist': case 'nrc_rr_inv_hist':
 196                  $dt = $this->getDateStr(array('I_Inv_Dt'),$model); break;
 197              case 'nrc_xfer_inv_hist':
 198                  $dt = $this->getDateStr(array('Tr_Date'),$model); break;
 199              }
 200          $query = '';
 201          switch($this->type)
 202              {
 203              case 'nrc_adv_pay_inv_hist':
 204                  $query = "SELECT Domain, Holder, RegDate, RenDate, Amount, VAT, Total, I_Inv_No FROM nrc_adv_pay_inv_hist WHERE I_Bill_Nh ='".$u. "' AND ".$dt;
 205                  break;
 206              case 'nrc_rr_inv_hist':
 207                  $query = "SELECT I_Inv_No, Domain, Holder, RegDate, RenDate, Amount, VAT FROM nrc_rr_inv_hist WHERE I_Bill_Nh ='".$u."'  AND ".$dt;
 208                  break;
 209              case 'nrc_xfer_inv_hist':
 210                  $query = "SELECT Domain, Holder, RegDate, RenDate, Amount, VAT, Total, I_Inv_No, Tr_Date FROM nrc_xfer_inv_hist WHERE I_Bill_Nh ='".$u."'  AND ".$dt;
 211                  break;
 212              case 'XferTo':
 213                  $query = "SELECT Domain, D_Holder, Moved_Dt, D_Reg_Dt, D_Ren_Dt FROM nrc_xfers_list WHERE New_Nic_Handle ='".$u."' AND ".$dt;
 214                  break;
 215              }
 216          if($criteria!=null and $criteria != '') $query .= ' AND ('. preg_replace('/^AND/', ' ', $criteria).')';
 217          return $query;
 218          }

 230  	public function getDetails($model,$criteria=null)
 231          {
 232          Yii::log(print_r($model,true), 'debug', 'Accounts_history_invoicesController::getDetails');
 233          $cache_key = 'Accounts_history_invoicesController_getDetails_'.print_r($model,true).'_'.print_r($criteria,true);
 234          $d = Yii::app()->cache->get($cache_key);
 235          if($d==null)
 236              {
 237              $d = $this->uncached_getDetails($model,$criteria);
 238              $timeout = 20; // 20 secs
 239              Yii::app()->cache->set($cache_key, $d, $timeout);
 240              }
 241          Yii::log($d, 'debug', 'Accounts_history_invoicesController::getDetails');
 242          return $d;
 243                  }

 255  	protected function uncached_getDetails($model,$criteria=null)
 256          {
 257          $query = $this->getPartialQuery($model,$criteria);
 258          $connection = Yii::app()->db;
 259          $command = $connection->createCommand('SELECT I_Inv_No, sum(COALESCE(VAT,0)) as theVat, sum(COALESCE(Amount,0)) as theAmount from ('.$query.') x GROUP BY I_Inv_No');
 260          $results = $command->queryAll(); 
 261  
 262          foreach($results as $aRow)
