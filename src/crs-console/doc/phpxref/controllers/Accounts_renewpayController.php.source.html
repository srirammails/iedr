 211      /**
 212       * performs the fund deposit function via CRS-WS-API, which uses RealEx from the java middle tier
 213       * 
 214       * @param  AccountTopUpModel   $model data model
 215       * @return boolean   true on success
 216       * @access protected
 217       * @see    https://resourcecentre.realexpayments.com/products.html?id=124
 218       */
 219  	protected function deposit_funds($model)
 220          {
 221          $result = null;
 222          $payreq = CreditCardFormSegmentModel::asWSAPIPaymentObject($model);
 223          $ret = CRSPaymentAppService_service::depositFunds($result,$this->backend_errors, Yii::app()->user->authenticatedUser, $payreq);
 224          if(!$ret)
 225              // errs may contain 'AccessDeniedException','PaymentException',
 226              Yii::log(print_r($this->backend_errors,true), 'error', 'Accounts_renewpayController::deposit_funds()');
 227          else
 228              {
 229              Yii::log(print_r($result,true), 'debug', 'Accounts_renewpayController::deposit_funds() : SUCCESS');
 230              // invalidate cached value
 231              invalidate_DepositBalance(Yii::app()->user->authenticatedUser->username);
 232              }
 233          return $ret;
 234          }

 236      /**
 237       * returns sql fragment to the parent class for grid-data fetching
 238       * 
 239       * @param  OfflineBatchSelectionModel  $model    selection model
 240       * @param  string $criteria search criteria
 241       * @return string sql query
 242       * @access public
 243       */
 244  	public function getPartialQuery($model,$criteria)
 245          {
 246          $u = Yii::app()->user->name;
 247          $query = '';
 248          switch($this->type)
 249              {
 250              case 'outstanding_payments':
 251                  $query = 'select * from nrc_mngoff_sessions where Nic_Handle=\''.$u.'\'';
 252                  break;
 253              case 'view_batch':
 254                  $query = 'select p.*, d.D_Holder from PendingPayment p, Domain d where p.D_Name=d.D_Name AND Sess_ID = \''.$model->session.'\'';
 255                  break;
 256              }
 257          if($criteria!=null and $criteria != '') $query .= ' AND ('. preg_replace('/^AND/', ' ', $criteria).')';
 258          return $query;
 259          }
