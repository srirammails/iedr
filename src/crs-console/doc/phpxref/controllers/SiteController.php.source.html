 101  	public function actionSummary()
 102          {
 103          $billing_accno = Yii::app()->user->id;
 104          $billing_nic   = Yii::app()->user->name;
 105          switch($_GET['id'])
 106              {
 107              case 'DepositBalance':        echo number_format(get_stats_DepositBalance($billing_nic),2); break;
 108              case 'DomainCount':        echo number_format(get_stats_DomainCount($billing_accno)); break;
 109              case 'NewRegs':            echo get_stats_NewRegs($billing_accno); break;
 110              case 'PassedTickets':        echo get_stats_PassedTickets($billing_accno); break;
 111              case 'PendingTickets':        echo get_stats_PendingTickets($billing_accno); break;
 112              case 'StatementBalance':    echo number_format(get_stats_StatementBalance($billing_nic),2); break;
 113              case 'TicketCount':        echo get_stats_TicketCount($billing_accno); break;
 114              case 'TransfersIn':        echo get_stats_TransfersIn($billing_nic); break;
 115              case 'TransfersOut':        echo get_stats_TransfersOut($billing_nic); break;
 116              }

 132              {
 133                  $headers="From: {$model->email}\r\nReply-To: {$model->email}";
 134                  mail(Yii::app()->params['adminEmail'],$model->subject,$model->body,$headers);
 135                  Yii::app()->user->setFlash('contact','Thank you for contacting us. We will respond to you as soon as possible.');
 136                  $this->refresh();
 137              }
 138          }
 139          $this->render('contact',array('model'=>$model));

 143       * Displays the login page, or handles a login request - on success, redirecting to originating url, if any
 144       *
 145           * @access public
 146           * @see    LoginForm, login.php, NicHandleIdentity
 147       */
 148  	public function actionLogin()
 149      {
 150          $model=new LoginForm();
 151  
 152          // if it is ajax validation request
 153          if(isset($_POST['ajax']) && $_POST['ajax']==='login-form')
 154          {
 155              echo CActiveForm::validate($model);
 156              Yii::app()->end();
 157          }
 158  
 159          // collect user input data
 160          if(isset($_POST['LoginForm']))
 161          {
 162              $model->attributes=$_POST['LoginForm'];
 163              // validate user input and redirect to the previous page if valid
 164              if($model->validate() && $model->login())
 165                  $this->redirect(Yii::app()->user->returnUrl);
 166          }
 167          // display the login form
 168          $this->render('login',array('model'=>$model));
 169      }
