 123      /**
 124       * renders confirmation page
 125       * 
 126       * @return void     
 127       * @access protected
 128       * @see    Accounts_msdController::actionConfirm(), accounts_msd/confirm_action.php, MSDSelectionModel
 129       */
 130  	protected function showConfirmPage()
 131          {
 132          $postedForm = is_array($_POST['gridactionpay']) ? $_POST['gridactionpay'] : ( is_array($_POST['gridactionnopay']) ? $_POST['gridactionnopay'] : null );
 133          if($postedForm != null)
 134              {
 135              // TODO validate request is coming from a grid-commmand-confirm with a domainlist, action, confirmation
 136              $msd_model = $this->getSelectionModel();
 137              $msd_model->setFromPOST($postedForm);
 138              switch($msd_model->command)
 139                  {
 140                  case 'payonline':
 141                      $msd_model->needCreditCard=1;
 142                      break;
 143                  case 'paydeposit': case 'reactivate': default:
 144                      $msd_model->needCreditCard=0;
 145                      break;
 146                  }
 147  
 148              Yii::log('redirect to confirmation page, with serialized model='.print_r($msd_model,true), 'debug', 'Accounts_msdController::showConfirmPage()');
 149              $this->redirect(array('accounts_msd/confirmaction','id'=>self::safeSerializeObj($msd_model) ));
 150              }
 151          else
 152              {
 153              Yii::log('not a POST["gridactionpay"] or POST["gridactionnopay"] request', 'error', 'Accounts_msdController::showConfirmPage()');
 154              $this->redirect($msd_model->returnurl,true);
 155              }
 156          }

 243      /**
 244       * Get Deleted-in-last-year Grid data with search and sort parameters; return Array.
 245       * 
 246       * @param  object    $model    data model containing column model and criteria
 247       * @param  object    $criteria search criteria
 248       * @param  object    $sort     sort criteria
 249       * @param  integer   $perpage  rows per page, used to calculate 1st row of 'pagenum' page
 250       * @param  integer   $pagenum  pagenum to return data for
 251       * @return array     array containing metadata and row array
 252       * @access protected
 253       */
 254  	protected function getDeletedData($model,$criteria,$sort,$perpage,$pagenum)
 255          {
 256          // paging
 257          $pagedata    = array('records'=>0,'page'=>$pagenum,'total'=>0,'rows'=>array());
 258          $offset     = $perpage * ($pagenum-1);
 259          $limit      = $perpage;
 260          // API SOAP call
 261          $result     = new CRSDomainAppService_deletedDomainsDataVO();
 262  
 263          CRSDomainAppService_service::getDeletedDomains($result,$this->backend_errors, Yii::app()->user->authenticatedUser, Yii::app()->user->nicHandle->nicHandleId, $offset, $limit);
 264          if(count($this->backend_errors)==0)
 265              {
 266              Yii::log('getDeletedDomains results : '.print_r($result,true), 'warning', 'Accounts_msdController::getDeletedData()'.__LINE__);
 267              // records = total records on all pages
 268              $pagedata['records']    = $result->total;
 269              // total = total num result pages = (t div p)+1
 270              $pagedata['total']    = math_div($pagedata['records'],$perpage)+1;
 271              $pagedata['rows']       = array();
 272              if(isset($result->deletedDomains) and is_array($result->deletedDomains) and is_object($result->deletedDomains[0]))
 273                  foreach($result->deletedDomains as $o) {
 274                      Yii::log('calling addResult : '.print_r($o,true), 'debug', 'Accounts_msdController::getDeletedData()');
 275                      $r = $this->addResult($model,$o);
 276                      if($r != null) $pagedata['rows'][] = $r;
 277                      }
 278              else
 279                  if(is_object($result) and $result->deletedDomains)
 280                      {
 281                      $r = $this->addResult($model,$result->deletedDomains);
 282                      if($r != null) $pagedata['rows'][] = $r;
 283                      }
 284                  else
 285                      Yii::log('no results : '.print_r($result,true), 'warning', 'Accounts_msdController::getDeletedData()'.__LINE__);
 286              }
 287          else Yii::log(print_r($this->backend_errors,true), 'error', 'Accounts_msdController::getDeletedData()'.__LINE__);
 288  
 289          Yii::log(print_r($pagedata,true), 'debug', 'Accounts_msdController::getDeletedData()');
 290  
 291          return $pagedata;
 292          }

 294      /**
 295       * Get Recent-MSD Grid data with search and sort parameters; return Array.
 296       * 
 297       * @param  object    $model    data model containing column model and criteria
 298       * @param  object    $criteria search criteria
 299       * @param  object    $sort     sort criteria
 300       * @param  integer   $perpage  rows per page, used to calculate 1st row of 'pagenum' page
 301       * @param  integer   $pagenum  pagenum to return data for
 302       * @return array     array containing metadata and row array
 303       * @access protected
 304       */
 305  	protected function getInvoicesData($model,$criteria,$sort,$perpage,$pagenum)
 306          {
 307          $noPendingPayment = true;
 308          // paging
 309          $pagedata    = array('records'=>0,'page'=>$pagenum,'total'=>0,'rows'=>array());
 310          $offset     = $perpage * ($pagenum-1);
 311          $limit      = $perpage;
 312          // API SOAP call
 313          $result     = new CRSPaymentAppService_invoiceVO();
 314          $invSummary = $this->getMSDInvoicesSummary($criteria);
 315          $count = $invSummary->count;
 316          if($count != null and $count > 0)
 317              {
 318              /*need for reliability, api breaks if 'nil' xml tag sent */ // (TODO NASK fixed?) $criteria->billingStatus=null; unset($criteria->billingStatus); // NASK#4246
 319              Yii::log('crit : '.print_r($criteria,true), 'debug', 'Accounts_msdController::getInvoicesData()');
 320  
 321              CRSPaymentAppService_service::getAllMsdDomains($result,$this->backend_errors, Yii::app()->user->authenticatedUser, $criteria, $offset, $limit, $sort);
 322              if(count($this->backend_errors)==0)
 323                  {
 324                  // records = total records on all pages
 325                  $pagedata['records']    = $count;
 326                  // total = total num result pages = (t div p)+1
 327                  $pagedata['total']    = math_div($count,$perpage)+1; // BUG when numrows is exact multiple of pagesize, this goes off by 1 ; check all other instances!
 328                  if(is_array($result) and is_object($result[0]))
 329                      {
 330                      foreach($result as $k => $o) {
 331                          #Yii::log('calling addResult : '.print_r($o,true), 'debug', 'Accounts_msdController::getInvoicesData()');
 332                          $r = $this->addResult($model,$o);
 333                          if($r != null) $pagedata['rows'][] = $r;
 334                          }
 335                      }
 336                  else
 337                      if(is_object($result))
 338                          {
 339                          $r = $this->addResult($model,$result);
 340                          if($r != null) $pagedata['rows'][] = $r;
 341                          }
 342                      else
 343                          Yii::log('no results : '.print_r($result,true), 'warning', 'Accounts_msdController::getInvoicesData()'.__LINE__);
 344                  }
 345              else Yii::log(print_r($this->backend_errors,true), 'error', 'Accounts_msdController::getInvoicesData()'.__LINE__);
 346              }
 347          return $pagedata;
 348          }
