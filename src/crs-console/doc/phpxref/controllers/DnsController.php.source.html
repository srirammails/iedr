 181  
 182          $vad = AuthOnlyBaseController::safeDeserializeObj($_REQUEST['id']);
 183          $vad->attributes=$_POST['DNS_BulkMod'];

 185          if(substr_count($vad->domainlist, ",")>0){
 186              $vad->ip = array();        
 187          }
 188  
 189          if($vad->validate()){#Only do this if the model validates?
 190              $thearray = explode(",", $vad['domainlist']);
 191  
 192              foreach($thearray as $domArr){
 193                  foreach($vad['ns'] as $key=>$ns){
 194                      if($ns){
 195                          if($vad->ip[$key]){ $ns = $vad->ip[$key]; }#If glue has been provided, use it in place of the nameserver,
 196                          exec(  Yii::app()->params['ckdnsPath'] . " " . escapeshellarg("$ns $domArr")  , $return_message_array, $return_number);
 197  
 198                          if ($return_number != 0){
 199                              $str .= "Problem executing ckdns, $errStatement";
 200                              array_push($vad->message, $str);
 201                              $this->render('dns/dnsupdate',array('model'=>$vad));
 202                              exit();                                                    
 203                          }
 204                          else{
 205                              $VerifiedResult = end($return_message_array);
 206                              $str = "Verification result for $domArr on $ns....";    
 207                              Yii::log("\nVerification result for $domArr on $ns....");
 208  
 209                              if (eregi('FATAL', $VerifiedResult)){
 210                                  $bad[] = $domArr; 
 211                                  $str.="Invalid";
 212                                  Yii::log("Invalid\n");
 213                              }
 214                              else{
 215                                  $good[] = $domArr; 
 216                                  $str .="Valid";
 217                                  Yii::log("valid\n");
 218                              }
 219                          }
 220                          array_push($vad->message, $str);
 221                      }
 222                  }
 223              }
 224  
 225              #if any fail is detected, then the batch is not processed.  For testing, just allow changes regardless.  Need to set this to !$count for production.
 226              if(count($bad)){                
 227                  if(!$this->saveDNS($vad)){
 228                      array_push($vad->message, "API Failure - update not applied,  $errStatement");
 229                  }
 230              }
 231              else{
 232                  array_push($vad->message, "Errors found in the DNS validation - update not applied,  $errStatement");
 233              }
 234          }

 263                  $remark = 'DNS Update via CRS-WS-API';
 264  
 265          foreach($doms as $aDom){
 266              CRSDomainAppService_service::save($this->backend_errors, $user, $aDom, "", "", "", "", "", "", $ns_arr, "", $remark);        
 267          }
 268  
 269          Yii::log('errors="'.print_r($this->backend_errors,true).'"', 'debug', 'DnsController::saveDNS()');
