 108              $et_model->errors = $this->backend_errors;

 148      /**
 149       * Perform ticket update
 150       * 
 151       * Ticket update via CRS-WS-API must:
 152       * 
 153       * - Fetch ticket in 'edit' mode
 154       * 
 155       * - 'Checkout' ticket, to lock it for updates
 156       * 
 157       * - reformat data changes as flattened domain-operations array {@link create_ticket_operations_from_submitted_data}
 158       * 
 159       * - Update ticket data
 160       * 
 161       * - 'Checkin' ticket, to unlock
 162       * 
 163       * @param  CRSTicketAppService_ticketVO   $ticketVO Ticket data from CRS-WS-API
 164       * @param  EditTicketModel    $et_model ticket edit data
 165       * @return boolean   true on success
 166       * @access protected
 167       */
 168  	protected function doUpdateTicket($ticketVO,$et_model)
 169          {
 170          global $admin_sts_wsapi_strings;
 171          # proceed to update ticket
 172          $ticket_id = $et_model->id;
 173          $user = Yii::app()->user->authenticatedUser;
 174          # attempt to fetch for edit (possibly not authorized)
 175          CRSTicketAppService_service::edit($ticketVO,$this->backend_errors, $user, $ticket_id);
 176          if($this->backend_errors==null)
 177              {
 178              if($this->check_out_ticket($user,$ticketVO))
 179                  {
 180                  # modify data & update
 181                  $domainOperations = $this->create_ticket_operations_from_submitted_data($ticketVO,$et_model);
 182                  sleep(1); # NB: need to sleep at least 1 sec, so TicketHist primary key (includes Change-date) will be unique
 183                  CRSTicketAppService_service::update($this->backend_errors, $user, $ticket_id, $domainOperations, $et_model->requestersRemark, $et_model->clikPaid);
 184                  if(count($this->backend_errors)==0)
 185                      {
 186                      sleep(1); # NB: need to sleep at least 1 sec, so TicketHist primary key (includes Change-date) will be unique
 187                      CRSTicketAppService_service::checkIn($this->backend_errors, $user, $ticket_id, $admin_sts_wsapi_strings[5]);
 188                      if($this->backend_errors==null)
 189                          return true;
 190                      }
 191                  else $this->backend_errors .= ': failed update';
 192                  }
 193              else $this->backend_errors .= ': failed checkout';
 194              }
 195          return false;
 196          }

 198      /**
 199       * performs ticket checkout, returning true on success.
 200       * 
 201       * @param  CRSTicketAppService_authenticatedUserVO    $user     user object to check out to
 202       * @param  CRSTicketAppService_ticketVO    $ticketVO ticket object (only 'id' attribute is used)
 203       * @return boolean   true on success
 204       * @access protected
 205       */
 206  	protected function check_out_ticket($user,$ticketVO) {
 207          $checkedout = false;
 208          if($ticketVO->checkedOutTo == $user->username)
 209              $checkedout = true;
 210          else
 211              {
 212              sleep(1); # NB: need to sleep at least 1 sec, so TicketHist primary key (includes Change-date) will be unique
 213              # attempt to check out
 214              CRSTicketAppService_service::checkout($this->backend_errors, $user, $ticketVO->id);
 215              if(count($this->backend_errors)==0)
 216                  $checkedout = true;
 217              }
 218          return $checkedout;
 219          }

 221      /**
 222       * transforms the supplied objects into an array of changes
 223       * 
 224       * @param  object    $ticketVO          Ticket object as fetch frmo CRS-WS-API with Ticket-ID
 225       * @param  object    $edit_ticket_model Vaildated Web Form Data of Ticket changes
 226       * @return array     array of data from merging ticket-object with web form ticket changes
 227       * @access protected
 228       */
 229  	protected function create_ticket_operations_from_submitted_data($ticketVO,$edit_ticket_model) {
 230          $domainOperations = array();
 231  
 232          # extract class/category from applicant
 233          $clascat = explode(':',$edit_ticket_model->applicant);
 234          $edit_ticket_model->operation_domainHolderClassField_newValue = $clascat[0];
 235          $edit_ticket_model->operation_domainHolderCategoryField_newValue = $clascat[1];
 236          # add change operations : iterate over usual change-fields
 237          $opsarr    = array('domainHolderField','domainHolderClassField','domainHolderCategoryField','domainNameField','resellerAccountField',);
 238          foreach($opsarr as $op)
 239              {
 240              $fld = 'operation_'.$op.'_newValue';
 241              $val = isset($edit_ticket_model->$fld) ? $edit_ticket_model->$fld : $ticketVO->operation->$op->newValue ;
 242              if($val!=null and $val !='')
 243                  $domainOperations[$op] = array(
 244                      'newValue'        => $val,
 245                      'failureReasonId'    => 0+$ticketVO->operation->$op->failureReasonId,
 246                      );
 247              }
 248          # tech and billing contacts
 249          $opsarr    = array('techContactsField','billingContactsField');
 250          foreach($opsarr as $op)
 251              {
 252              $fld = 'operation_'.$op.'_newValue';
 253              $val = isset($edit_ticket_model->$fld) ? $edit_ticket_model->$fld : $ticketVO->operation->$op->newValue->nicHandle ;
 254              if($val!=null and $val !='')
 255                  $domainOperations[$op] = array(
 256                      'newValue'        => $val,
 257                      #'failureReasonId'    => 0+$ticketVO->operation->$op->failureReasonId,
 258                      );
 259              }
 260  
 261          # add change operations : iterate over admin contacts
 262          if(!is_array($ticketVO->operation->adminContactsField))
 263              {
 264              $t = $ticketVO->operation->adminContactsField;
 265              $ticketVO->operation->adminContactsField = array();
 266              $ticketVO->operation->adminContactsField[0] = $t;
 267              }
 268          $domainOperations['adminContactsField'] = array();
 269          foreach(array(0,1) as $x)
 270              {
 271              $op = $ticketVO->operation->adminContactsField[$x];
 272              $fld = 'operation_adminContactsField_'.$x.'_newValue_nicHandle';
 273              $val = isset($edit_ticket_model->$fld) ? $edit_ticket_model->$fld : $op->newValue->nicHandle ;
 274              if($val != null and $val != '')
 275                  $domainOperations['adminContactsField'][$x] = array(
 276                      'newValue'        => $val,
 277                      'failureReasonId'    => 0+($op->newValue->failureReasonId),
 278                      );
 279              }
 280          # add change operations : iterate over domains
 281          foreach(array(0,1,2) as $x)
 282              {
 283              $op = $ticketVO->operation->nameservers[$x];
 284              # name
 285              $fld_n = 'operation_nameservers_'.$x.'_name_newValue';
 286              $val_n = $edit_ticket_model->$fld_n;
 287              # ip address
 288              $fld_a = 'operation_nameservers_'.$x.'_ipAddress_newValue';
 289              $val_a = $edit_ticket_model->$fld_a;
 290  
 291              if($val_n!=null and $val_n!='')
 292                  {
 293                  $a = array(
 294                      'name'  => array(
 295                          'newValue'        => $val_n,
 296                          'failureReasonId'    => 0), // 0+$op->name->failureReasonId),
 297                      'ipAddress' => array(
 298                          'newValue'        => $val_a,
 299                          'failureReasonId'    => 0), // 0+$op->ipAddress->failureReasonId),
 300                      );
 301                  if($val_a==null or $val_a='')
 302                      unset($a['ipAddress']);
 303                  $domainOperations['nameservers'][] = $a;
 304                  }
 305              }
 306          Yii::log(print_r($domainOperations,true), 'debug', 'TicketsController::create_ticket_operations_from_submitted_data()');
 307          return $domainOperations;
 308          }

 329      /**
 330       * returns a pagefull of data for use by the parent GridController
 331       * 
 332       * @param  object    $model    model containing jqGrid colmodel-definitions
 333       * @param  object    $criteria search criteria
 334       * @param  unknown   $sort     sort criteria (unused)
 335       * @param  integer   $perpage  record per page
 336       * @param  integer   $pagenum  pagenumber to show
 337       * @return array     array of meta-data and data for a page of data for the ticket grid display
 338       * @access protected
 339       */
 340  	protected function getData($model,$criteria,$sort,$perpage,$pagenum)
 341          {
 342          Yii::log('..', 'debug', 'TicketsController::getData($model,$criteria,$sort,$perpage,$pagenum)'.__LINE__);
 343          // paging
 344          $pagedata    = array('records'=>0,'page'=>$pagenum,'total'=>0,'rows'=>array());
 345          $offset        = $perpage * ($pagenum-1);
 346          $limit        = $perpage;
 347  
 348          if(!$model->days){ $model->days = 100; }
 349          $tm = time();
 350          $criteria->from = date('Y-m-d',$tm-(86400*$model->days)).'T00:00:00';
 351          $criteria->to   = date('Y-m-d',$tm).'T00:00:00';
 352  
 353          $sc = new CRSTicketAppService_sortCriterion();
 354          $sc->ascending    = true;// type=xs:boolean, (NOT NULL), min=1
 355          if($_GET['sord'] == "desc"){   $sc->ascending    = false;   }
 356  
 357          if(is_null($model->columns[$_GET['sidx']]['criteriafield'])){
 358              $sc->sortBy    = 'id';// type=xs:string, (NOT NULL), min=0
 359          }
 360          else{
 361              $sc->sortBy = $model->columns[$_GET['sidx']]['criteriafield'];
 362          }
 363  
 364          // API SOAP call
 365          $result     = null;
 366          CRSTicketAppService_service::find($result,$this->backend_errors, Yii::app()->user->authenticatedUser, $criteria, $offset, $limit, $sc);
 367          if(count($this->backend_errors)==0)
 368              {
 369              // records = total records on all pages
 370              $pagedata['records']     = $result->totalResults;
 371              // total = total num result pages = (t div p)+1
 372              $pagedata['total']         = math_div($result->totalResults,$perpage)+1;
 373              if(is_array($result->list))
 374                  foreach($result->list as $k => $o)
 375                      $pagedata['rows'][] = $model->addResults($o);
 376              else
 377                  if(is_object($result->list))
 378                      $pagedata['rows'][] = $model->addResults($result->list);
 379                  else
 380                      Yii::log(print_r($result,true), 'warning', 'TicketsController::getData() : WARN : no \'results\':'.__LINE__);
 381              }
 382          else Yii::log(print_r($this->backend_errors,true), 'error', 'TicketsController::getData() : ERROR:'.__LINE__);
 383          return $pagedata;
 384          }
