  39      /**
  40       * Domain Registration Form
  41       * 
  42       * Has 4 modes; 
  43       *
  44       *  - without request parameters, displays empty domain-registration form
  45       *
  46       *  - validates domain-registration form data Ajax requests
  47       *
  48       *  - on receipt of invalid form data, redisplays form with error messages
  49       *
  50       *  - on receipt of valid form data, redirects to domain-registration-details form
  51       *
  52       * Domains are checked for availability as part of validation.
  53       * Availability-checking includes checking whether the domain already exists, and whether there already exists a registration-ticket for the domain name. See {@link domainIsAvailable}.
  54       * The redirected-to domainDetails form/action is the one that performs the actual domain creation
  55       *
  56       * @return void  
  57       * @access public
  58       * @see    regnew.php, Domains_RegNew
  59       */
  60  	public function actionRegnew()
  61          {
  62          $errCount = 0;
  63          $rn_model = new Domains_RegNew();
  64          if(Yii::app()->request->isPostRequest)
  65              {
  66              $this->performAjaxValidation($rn_model);
  67              if(isset($_POST['Domains_RegNew']))
  68                  {
  69                  $rn_model->attributes=$_POST['Domains_RegNew'];
  70                  if($rn_model->validate())
  71                      {
  72                      Yii::log('model validates ok', 'debug', 'DomainsController::actionRegnew()');
  73                      $val_avail_domains = array();
  74                      $allDomains = $rn_model->getDomainsAsArray();
  75                      foreach($allDomains as $domain)
  76                          {
  77                          // check if this domain is available
  78                          if(Utility::domainIsAvailable($domain)){
  79                              $val_avail_domains[] = $domain;
  80                              }
  81                          else{
  82                              $errCount++;
  83                              $rn_model->addError('domain_names', $domain.' is invalid, already registered, or has a pending ticket.');    
  84                              }
  85                          }
  86                      if(count($val_avail_domains) == count($allDomains))
  87                          {
  88                          // go to next step : enter domain details
  89                          Yii::log('go to next step : enter domain details, with domain names: '.print_r($val_avail_domains,true), 'debug', 'DomainsController::actionRegnew()');
  90                          $vad = AuthOnlyBaseController::safeSerializeObj($val_avail_domains);
  91                          Yii::log('domain names serialized: '.$vad, 'debug', 'DomainsController::actionRegnew()');
  92                          $this->redirect(array('domaindetails','validavailabledomains'=>$vad,),true);
  93                          }
  94                      }
  95                  }
  96              }
  97          // display form (with any errs)
  98          $this->render('regnew',array('model'=>$rn_model));
  99          }

 120  	public function actionDomainDetails()
 121          {
 122          $dd_model = new Domains_Details();
 123          if(Yii::app()->request->isPostRequest)
 124              {
 125              // POST : if valid, then create, else redisplay
 126              if(isset($_POST['Domains_Details']))
 127                  {
 128                  $this->performAjaxValidation($dd_model);
 129                  $dd_model->attributes=$_POST['Domains_Details'];
 130                  //Yii::log('POST : '.print_r($_POST,true), 'debug', 'DomainsController::actionDomainDetails()');
 131                  //Yii::log('(CREATE)dd_model : '.print_r($dd_model,true), 'debug', 'DomainsController::actionDomainDetails()');
 132                  if($dd_model->validate())
 133                      {
 134                      $dc_model = new Domains_Created($this->createDomains($dd_model));
 135                      $smdl = AuthOnlyBaseController::safeSerializeObj($dc_model);
 136                      $this->redirect(array('domainscreated','domaincreateresults'=>$smdl),true);
 137                      }
 138                  }
 139              if(isset($_POST['Nichandle_Details']) and isset($_POST['Nichandle_Details']['returningData']))
 140                  {
 141                  // returning from a NIC find/edit/create
 142                  $returndata = $_POST['Nichandle_Details']['returningData'];
 143                  $dd_model = AuthOnlyBaseController::safeDeserializeObj($returndata['formdata']);
 144                  $nicrole = $returndata['nictype'];
 145                  $dd_model->$nicrole = $returndata['nic'];
 146                  //Yii::log('(RETURN)dd_model : '.print_r($dd_model,true), 'debug', 'DomainsController::actionDomainDetails()');
 147                  //Yii::log('returned from another form : set model nic "'.$nicrole.'" to "'.$nic.'"', 'debug', 'DomainsController::actionDomainDetails()');
 148                  //Yii::log('returned from another form : '.print_r($returndata,true), 'debug', 'DomainsController::actionDomainDetails()');
 149                  }
 150              }
 151          if(isset($_GET['validavailabledomains']) and (($dd_model->domainlist==null) or (count($dd_model->domainlist)<1)))
 152              {
 153              // got here from post from RegNew (or unexpected direct access)
 154              $vad_packed = $_GET['validavailabledomains'];
 155              $validavailabledomains = AuthOnlyBaseController::safeDeserializeObj($vad_packed);
 156              //Yii::log('validavailabledomains='.print_r($validavailabledomains,true), 'debug', 'DomainsController::actionDomainDetails()');
 157              $dd_model = new Domains_Details($validavailabledomains);
 158              //Yii::log('(FROMREGNEW)dd_model : '.print_r($dd_model,true), 'debug', 'DomainsController::actionDomainDetails()');
 159              // back to domain-select page if no domains listed
 160              if(($dd_model->domainlist==null) or (count($dd_model->domainlist)<1))
 161                  $this->redirect('domains/regnew',true);
 162              }
 163  
 164          // error/fallback : redisplay form with errors
 165          $this->render('domaindetails',array('model'=>$dd_model));
 166          }

 202  	public function actionViewDomain()
 203          {
 204          $vd_model = new ViewDomainModel();
 205          $domainName = "";
 206          $str = "";
 207          $results = null;
 208  
 209          if(Yii::app()->request->isPostRequest)
 210              {
 211              switch($_POST['yt0'])
 212                  {
 213                                  case 'Send Domain to MSD':
 214                      $domainName = $_POST['ViewDomainModel']['domain_name'];
 215  
 216                      CRSDomainAppService_service::isMsd($results,$vd_model->errors, Yii::app()->user->authenticatedUser, $domainName);
 217                      if(!$results)
 218                          {        
 219                          CRSDomainAppService_service::moveToMsd($vd_model->errors,Yii::app()->user->authenticatedUser,$domainName);
 220                          $str = "Domain $domainName added to MSD";
 221                                  array_push($vd_model->message, $str);
 222                          }
 223                      else
 224                          {    
 225                          array_push($vd_model->message,"No action performed, $domainName already in MSD");
 226                          }
 227                                          break;
 228                  default :
 229                      $this->redirect('site/error','Update Not Allowed');
 230                      break;
 231                                  }
 232              }
 233          else
 234              {
 235              $domainName =  $_GET['id'];
 236              }
 237  
 238          $domainVO = $this->getDomainDetails($domainName);
 239          $vd_model->fillFromObject($domainVO, $this->backend_errors);
 240          $this->render('viewdomain',array('model'=>$vd_model));
 241          }

 274      /**
 275       * creates one or more domains, using the domain list and details of the supplied model
 276       * 
 277       * @param  Domains_Details     $model Domain Detail model, including domain list
 278       * @return array     map of domain-names, whose creation was attempted, to results array, containing either 'ticket' => ticket-number (on success) or 'error' => error string, on error
 279       * @access protected
 280       */
 281  	protected function createDomains($model)
 282          {
 283          $domain_creation_results = array();
 284  
 285          $regarr  = get_reg_prices();
 286          #Yii::log('regarr:'.print_r($regarr,true), 'debug', 'DomainsController::createDomains()');
 287          $r_codes = $regarr['CODE'];
 288          #Yii::log('r_codes:'.print_r($r_codes,true), 'debug', 'DomainsController::createDomains()');
 289          $r_obj   = $r_codes[$model->registration_period];
 290          #Yii::log('r_obj:'.print_r($r_obj,true), 'debug', 'DomainsController::createDomains()');
 291          $reg_period_years = $r_obj->duration;
 292          #Yii::log('reg_period_years:'.$reg_period_years, 'debug', 'DomainsController::createDomains()');
 293  
 294          // ticketCreateVO
 295          $data = new CRSTicketAppService_ticketCreateVO();
 296          // .. static values
 297              $data->type='REG';                    // type=tns:domainOperationType, (NOT NULL), min=0
 298              $data->resellerAccountId = Yii::app()->user->id;    // type=xs:long, (NOT NULL), min=1
 299              $data->hostmastersRemark = null;             // type=xs:string, (NOT NULL), min=0
 300              $data->adminStatus = 0;                    // type=xs:int, (NOT NULL), min=1
 301              $data->techStatus = 0;                    // type=xs:int, (NOT NULL), min=1
 302              $data->checkedOut = '';                    // type=xs:string, (NOT NULL), min=0
 303              $data->clikPaid = false;                // type=xs:boolean, (NOT NULL), min=1
 304          // .. from date
 305          $datenow = date('Y-m-d\TH:i:s',time());
 306          $datenowparts = explode('-',$datenow);
 307          $renewdate = ($datenowparts[0] + $reg_period_years).'-'.$datenowparts[1].'-'.$datenowparts[2];
 308              $data->adminStatusChangeDate = $datenow;        // type=xs:dateTime, (NOT NULL), min=0
 309              $data->techStatusChangeDate = $datenow;            // type=xs:dateTime, (NOT NULL), min=0
 310              $data->creationDate = $datenow;                // type=xs:dateTime, (NOT NULL), min=0
 311              $data->changeDate = $datenow;                // type=xs:dateTime, (NOT NULL), min=0
 312              $data->renewalDate = $renewdate;            // type=xs:dateTime, (NOT NULL), min=0
 313          // .. from model
 314          $class_cat_vals = explode(':',$model->applicant);
 315              $data->domainHolderClass = $class_cat_vals[0];        // type=xs:string, (NOT NULL), min=0
 316              $data->domainHolderCategory = $class_cat_vals[1];    // type=xs:string, (NOT NULL), min=0
 317              $data->domainHolder = $model->holder;            // type=xs:string, (NOT NULL), min=0
 318              $data->adminContact1NicHandle = $model->admin_contact_nic_1; // type=xs:string, (NOT NULL), min=0
 319              $data->adminContact2NicHandle = $model->admin_contact_nic_2; // type=xs:string, (NOT NULL), min=0
 320              if($data->adminContact2NicHandle == null or $data->adminContact2NicHandle=='')
 321                  unset($data->adminContact2NicHandle);
 322              $data->techContactNicHandle    = $model->tech_contact;    // type=xs:string, (NOT NULL), min=0
 323              $data->billingContactNicHandle = $model->billing_contact;     // type=xs:string, (NOT NULL), min=0
 324              $data->ns1 = $model->nameserver_name_1;            // type=xs:string, (NOT NULL), min=0
 325              $data->ns2 = $model->nameserver_name_2;            // type=xs:string, (NOT NULL), min=0
 326              $data->ns3 = $model->nameserver_name_3;            // type=xs:string, (NOT NULL), min=0
 327              $data->requestersRemark = $model->remarks;        // type=xs:string, (NOT NULL), min=0
 328          // creatorNicHandle is set with authentication credentials
 329  
 330          Yii::log('trying to create domains:'.print_r($model,true).', period='.$reg_period_years, 'debug', 'DomainsController::createDomains()');
 331  
 332          foreach($model->domainlist as $k => $domain)
 333              {
 334              $results = null;
 335              $data->domainName = $domain;
 336              // only send IP when nameserver matches hostname
 337              $match_ns_domain = '/\.'.$domain.'$/i';
 338              $data->ip1 = (preg_match($match_ns_domain, $data->ns1)==1) ? $model->nameserver_addr_1 : null;
 339              $data->ip2 = (preg_match($match_ns_domain, $data->ns2)==1) ? $model->nameserver_addr_2 : null;
 340              $data->ip3 = (preg_match($match_ns_domain, $data->ns3)==1) ? $model->nameserver_addr_3 : null;
 341              CRSTicketAppService_service::create($result, $this->backend_errors, Yii::app()->user->authenticatedUser, $data, $reg_period_years);
 342              if(count($this->backend_errors)==0)
 343                  $domain_creation_results[$domain]['ticket'] = $result;
 344              else
 345                  $domain_creation_results[$domain]['error'] = $this->backend_errors; 
 346              }
 347          return $domain_creation_results;
 348          }
