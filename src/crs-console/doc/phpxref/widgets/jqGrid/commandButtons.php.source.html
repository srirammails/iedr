  34      this.dpop = function (domain) { var n = this.list.indexOf(domain); if(n >= 0) { this.list.splice(n, 1); } };

  36      this.domainSelected = function (isChecked,domain) { if(isChecked) { this.dpush(domain); } else { this.dpop(domain); } };
