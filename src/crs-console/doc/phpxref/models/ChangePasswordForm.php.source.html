 105      /**
 106       * Function to check if the supplied old password actually is the same as the one stored.

 112  	public function matchesOldPasswordValidatorFn()
 113          {
 114  
 115          $storedPassword = "";
 116          $oldFormatPassword = "";
 117  
 118          $d = DB::get_db();
 119          $sql = "SELECT NH_Password FROM Access WHERE Nic_Handle = '" . Yii::app()->user->name . "' LIMIT 1;";
 120          $d->query($sql,__FILE__,__LINE__);
 121          while($r = $d->fetch())
 122          {
 123              $storedPassword = $r->NH_Password;
 124          }
 125  
 126          $sql = "SELECT OLD_PASSWORD('" . $this->old_password . "') AS NH_Password;";
 127          $d->query($sql,__FILE__,__LINE__);
 128          while($r = $d->fetch())
 129          {
 130              $oldFormatPassword = $r->NH_Password;
 131          }
 132  
 133          if(($oldFormatPassword != $storedPassword) and ($storedPassword != sha1($this->old_password))){
 134              $this->addError('old_password', 'Old password does not match stored password.');
 135          }
 136  
 137          return true;    
 138  
 139          }
