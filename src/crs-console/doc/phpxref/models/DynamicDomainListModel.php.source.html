 110  	public function domainListToArray()
 111          {
 112          // turn posted comma-separated list of domains into an array
 113          if($this->_domainlistArray==null or !is_array($this->_domainlistArray))
 114              {
 115              $this->_domainlistArray = array();
 116              $s = $this->domainlist;
 117              if($s!=null and strlen($s)>=4/*min valid domain name len*/)
 118                  {
 119                  if(!(strpos($s,',')===false))
 120                      foreach(explode(',',$s) as $k => $v)
 121                          $this->addDomainToList($v);
 122                  else $this->addDomainToList($s);
 123                  }
 124              // else : null, empty or invalid string - ignore
 125              }
 126          return $this->_domainlistArray;
 127          }

 187  	public function removeUncheckedDomains()
 188          {
 189          // remove domains from list(s) that are un-checked
 190          if($this->domainlist!=null)
 191              {
 192              $to_remove = array();
 193              foreach($this->domainListToArray() as $k => $v)
 194                  {
 195                  $mbrnam = DynamicDomainListModel::domainToVarPrefix($v);
 196                  $mbr_conf = $mbrnam.'_confirmed';
 197                  // test checkbox state
 198                  if(isset($this->$mbr_conf) and ($this->$mbr_conf)!=null)
 199                      if($this->$mbr_conf != 1)
 200                          $to_remove[$k] = $v;
 201                  }
 202              #Yii::log('removing domains : '.print_r($to_remove,true), 'error', 'DynamicDomainListModel::removeUncheckedDomains()');
 203              $this->setDomainListFromArray(array_diff($this->domainListToArray(), $to_remove));
 204              #Yii::log('remaining domains : '.print_r($this->domainlist,true), 'error', 'DynamicDomainListModel::removeUncheckedDomains()');
 205              }
 206          else
 207              Yii::log('domainlist not set', 'error', 'DynamicDomainListModel::removeUncheckedDomains()');

 218  	public function createDomainConfirmList()
 219          {
 220          // add model data members - to represent one checkbox and dropdownlist _per_domain_  - from a list of domains
 221          if($this->domainlist != null)
 222              {
 223              $regprods = get_reg_prices();
 224              $dft_prodcode = $regprods['DEFAULT']['REN'];
 225              foreach($this->domainListToArray() as $k => $v)
 226                  {
 227                  $mbrnam = DynamicDomainListModel::domainToVarPrefix($v);
 228  
 229                  // 'confirmed' for confirm checkbox
 230                  $mbr_conf = $mbrnam.'_confirmed';
 231                  $this->$mbr_conf = true;
 232                  $this->addValRule($mbr_conf,'boolean');
 233  
 234                  // 'product code' for drop-down renewal-period select-list
 235                  $mbr_prodcode = $mbrnam.'_prodcode';
 236                  $this->$mbr_prodcode = $dft_prodcode;
 237                  if($this->needRenewalOption==1)
 238                      $this->addValRule($mbr_prodcode,'ProductCodeValidator'); // any period
 239                  else
 240                      $this->addValRule($mbr_prodcode,'RenewalCodeValidator'); // 1-year only
 241  
 242                  // add domain-name as label for this one
 243                  $this->attr_labels[$mbr_conf] = $v;
 244                  }
 245              #Yii::log(print_r($this,true), 'debug', 'DynamicDomainListModel::createDomainConfirmList()');
 246              }
 247          else
 248              Yii::log('domainlist WAS NULL .. '.print_r($this,true), 'error', 'DynamicDomainListModel::createDomainConfirmList()');
 249          }
