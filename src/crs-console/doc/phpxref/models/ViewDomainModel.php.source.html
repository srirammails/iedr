 601      /**
 602       * copies attributes from an object instance into this model
 603       * 
 604       * @param  CRSDomainAppService_extendedDomainInfoVO  $obj  CRSDomainAppService_extendedDomainInfoVO instance to copy data from
 605       * @param  string $errs error string from domain retrieval, copied into model for view to display
 606       * @return void   
 607       * @access public 
 608       */
 609      public function fillFromObject($obj,$errs)
 610      {
 611      $flds = array(
 612          'tickets', 'documents', 'relatedDomainNames', 'pendingDomainNames',
 613          );
 614      foreach($flds as $f)
 615          if(isset($this->$f) and isset($obj->$f))
 616              $this->$f = $obj->$f;
 617      if(isset($obj->domain) and $obj->domain!=null)
 618          {
 619          $d = $obj->domain;
 620          addToObjectAsProperties($this,'domain',$d);
 621          if(isset($d->billingStatus) and $d->billingStatus!=null)
 622              {
 623              $billsts = get_billing_status();
 624              if($billsts!=null and is_array($billsts)) {
 625                  $b = $billsts[$d->billingStatus];
 626                  if($b!=null)
 627                      $this->domain_billingStatus = $b->DetailedDescription;
 628                  }
 629              }
 630          $this->domain_changeDate    = parseXmlDate($d->changeDate);
 631          # ? $this->domain_dateRoll    = parseXmlDate($d->dateRoll);
 632          $this->domain_domainStatusDate    = parseXmlDate($d->domainStatusDate);
 633          $this->domain_registrationDate    = parseXmlDate($d->registrationDate);
 634          $this->domain_renewDate        = parseXmlDate($d->renewDate);
 635          if(isset($d->resellerAccount))
 636              {
 637              $r = $d->resellerAccount;
 638              if(isset($r->changeDate))    $this->domain_resellerAccount_changeDate    = parseXmlDate($r->changeDate);
 639              if(isset($r->creationDate))    $this->domain_resellerAccount_creationDate    = parseXmlDate($r->creationDate);
 640              if(isset($r->statusChangeDate))    $this->domain_resellerAccount_statusChangeDate    = parseXmlDate($r->statusChangeDate);
 641              }
 642          if(isset($d->nameservers))
 643              $this->domain_nameservers = count($d->nameservers);
 644          }
 645      $this->errors = $errs; 
 646      Yii::log($this->domain_name.' -> '.$this->errors, 'debug', 'ViewDomainModel::fillFromObject()');
 647      }
