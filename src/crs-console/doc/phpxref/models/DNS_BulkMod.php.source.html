 206  	public function checkNS(){
 207          $errCount = 0;
 208          $count = 0;
 209  
 210          #Determine if duplicate NS labels were provided.
 211          if(count(array_filter($this->ns)) >  count(array_unique(array_filter($this->ns)))){
 212              $this->addError('ns','Duplicate Nameserver found.  Please remove the duplicate');
 213              $errCount++;

 215  
 216          #Remove the blank nameservers.
 217          $this->ns = array_filter($this->ns);
 218  
 219          #Take a look at the number of NS supplied.  There must be at least 2.
 220          if(count($this->ns) < 2){
 221               $this->addError('ns','Must supply at least 2 distinct nameservers.');
 222              $errCount++;
 223          }
 224  
 225          #Take a look at the syntax of the hostnames used for the nameserver labels.
 226          $hv = new HostNameValidator();
 227          foreach($this->ns as $key=>$aNS){
 228              if(!$hv->validateValue($aNS)){
 229                  $errCount++;
 230                  $this->addError('ns[' . $key . ']',"DNS syntax of Nameserver $key ($aNS) is invalid.");
 231              }
 232          }
 233  
 234          #If any errors were found.

 250  	public function validateDomainExistsOnAccount()
 251          {
 252          $allValid = true;
 253          foreach($this->getDomainsAsArray() as $dom)
 254              {
 255              $errArr = array();
 256              if(!Utility::domainExists($dom, Yii::app()->user->id))
 257                  {
 258                  $errArr[] = $dom;
 259                  }
 260              if(!$allValid)
 261                  $this->addError('domainlist','The domain(s) '.implode(', ',$errArr).' are not registered under your account, please remove.');
 262              }
 263          return $allValid;
