 189  	public function getSearch($searchparams)
 190          {
 191          $criteria = "";
 192          foreach($searchparams as $k => $r)
 193              {
 194              #Yii::log('searchparam '.print_r($k,true).' => '.print_r($r,true), 'debug', 'NicSearchModel::getSearch()');
 195              $f = $this->columns[$r->field]['criteriafield'];
 196              #Yii::log('criteriafield='.print_r($f,true), 'debug', 'NicSearchModel::getSearch()');
 197              if($f != null)
 198                  if($f=='Phone')
 199                      {
 200                      $phoneCriteria = preg_replace(array('/(P:|F:)/','/ *, */','/ *,+ */'),array('',',',','),$r->data);
 201                      $pieces = explode(',', $phoneCriteria);
 202                      foreach($pieces as $aPhone)
 203                          $criteria .= " OR Phone LIKE '%".$aPhone."%'";
 204                      }
 205                  else
 206                      $criteria .= " OR n.$f LIKE '%".$r->data."%'";
 207              }
 208          return preg_replace('/^ OR/', ' ', $criteria);
 209          }
