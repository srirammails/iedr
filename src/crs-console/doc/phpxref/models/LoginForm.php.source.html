  17   * LoginForm is the data structure for holding user-login-form data during a SiteController/login action request.

  19   * A Model Validator performs the authentication, by creating a CUserIdentity-subclass instance (NicHandleIdentity), and

 102              // password needs to be authenticated
 103              array('password', 'authenticate'),
 104              // internal username
 105              array('internalUser', 'validateInternalUser'),
 106              // array('internalUser', 'NicHandleValidator'), # no, would cause recursive nic lookups
 107              // verifyCode needs to be entered correctly
 108              array('verifyCode', 'AutoTestingCaptchaValidator', 'allowEmpty'=>!CCaptcha::checkRequirements()),

 141      /**
 142       * Validator function for internal-username form data
 143       * 
 144       * the internal-username must be found on the list returned by Utility::getIEDRUsers()

 151  	public function validateInternalUser($attrib,$params)
 152          {
 153          $v = ' is not a valid Internal User';
 154          $u = $this->$attrib;
 155          $usrs = Utility::getIEDRUsers();
 156          if(isset($usrs[$u]))
 157              $v = null;
 158          return $v;
 159          }

 161      /**
 162       * Validator function which authenticates username/password credentials
 163       * 
 164       * A NicHandleIdentity object instance is created with the request credentials,
 165       * then it's authenticate member function is called.
 166       * If authentication is not successful, an error message is added to the model
 167       * for the view to display.
 168       * 
 169       * @param  string  $attrib name of model-attribute being validated
 170       * @param  array   $params validation parameter array
 171       * @return string  null if valid, or error message string
 172       * @access public 
 173       */
 174  	public function authenticate($attribute,$params)
 175          {    
 176          if(!$this->hasErrors())
 177              {    
 178              $this->_identity=new NicHandleIdentity($this->username, $this->password);
 179              if(!$this->_identity->authenticate($internalUser))
 180                  $this->addError('password','Incorrect NIC-Handle or password.');
 181              }
 182          }
 183  
 184      /**
 185       * Logs in the user if authentication was successful
 186       * 
 187       * If authentication was not yet attempted, it is attempted here.
 188       * 
 189       * @return boolean whether login is successful
 190       * @access public 
 191       */
 192  	public function login()
 193          {
 194          if($this->_identity===null)
 195              {
 196              if(!($this->internalUser==NicHandleIdentity::$NOT_INTERNAL))
 197                  {
 198                  $this->username = $this->internalUser;
 199                  }
 200              $this->_identity = new NicHandleIdentity($this->username,$this->password);
 201              $this->_identity->authenticate();
 202              }
 203          if($this->_identity->errorCode===NicHandleIdentity::ERROR_NONE)
 204              {
 205              $duration=0; // prevent "cookie-based authentication"
 206              Yii::app()->user->login($this->_identity,$duration);
 207              return true;
 208              }
 209          else
 210              return false;
 211          }
 212      }
