 226  	protected function parseAndTrimPhones(&$a, $t)
 227          {
 228          $a = $this->parsePhoneNumbersIntoArray($t);
 229          foreach($a as $n => $p)
 230              {
 231              $tp = trim($p);
 232              if(strlen($tp) > 0)
 233                  $a[$n] = $tp;
 234              else
 235                  unset($a[$n]);
 236              }

 272      /**
 273       * validates array of phone/fax numbers
 274       * 
 275       * Before validation, the text-area contents have already been parsed into arrays; so we validate the array entries
 276       * 
 277       * @param  string  $attrib attribute name, either 'phones' or 'faxes'
 278       * @param  unknown $params validation parameters, not used
 279       * @return boolean false on any error
 280       * @access public 
 281       */
 282  	public function validatePhoneNumbersArray($attrib,$params)
 283          {
 284          // return string or null. On validation, the text-area contents have already been parsed into arrays; so we validate the array entries
 285          $ret = true;
 286          $attribarray = $attrib.'_array';
 287          if(is_array($this->$attribarray))
 288              {
 289              Yii::log($attrib.' '.print_r($this->$attribarray,true), 'debug', 'Nichandle_Details::validatePhoneNumbersArray() x');
 290              $lbls = $this->attributeLabels();
 291              foreach($this->$attribarray as $n => $p)
 292                  {
 293                  if(!(preg_match(PhoneNumberValidator::$phoneNumberRegex, $p)===1))
 294                      {
 295                      $lbls = $this->attributeLabels();
 296                      $lbl = $lbls[$attrib];
 297                      $msg = '"'.$p.'" is Not A Valid Phone Number';
 298                      $this->addError($attrib,Yii::t('yii',$msg));
 299                      $ret = false;
 300                      Yii::log($msg, 'warning', 'Nichandle_Details::validatePhoneNumbersArray()');
 301                      }
 302                  }
 303              }
 304          else Yii::log('Input Not An Array', 'error', 'Nichandle_Details::validatePhoneNumbersArray()');
 305          return $ret;
 306          }
