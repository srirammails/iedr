 123      /**
 124       * returns search criteria object for base class from request parameters
 125       * 
 126       * @param  array  $searchparams request parameters array
 127       * @return CRSTicketAppService_ticketSearchCriteria search criteria
 128       * @access public
 129       */
 130  	public function getSearch($searchparams)
 131          {
 132          $criteria = $this->getSearchBase();
 133          foreach($searchparams as $k => $r)
 134              {
 135              Yii::log('searchparam '.print_r($k,true).' => '.print_r($r,true), 'debug', 'AllTicketsModel::getSearch()');
 136              $f = $this->columns[$r->field]['criteriafield'];
 137              Yii::log('criteriafield='.print_r($f,true), 'debug', 'AllTicketsModel::getSearch()');
 138              if($f != null)
 139                  {
 140                  if($f=='adminStatus')
 141                      $criteria->$f = encode_admin_status_to_int($r->data);
 142                  else
 143                      if($f=='techStatus')
 144                          $criteria->$f = encode_tech_status_to_int($r->data);
 145                      else
 146                          if($f=='domainName' or $f=='domainHolder')
 147                              $criteria->$f = '%'.$r->data.'%';
 148                          else
 149                              $criteria->$f = $r->data;
 150                  }
 151              }
 152          Yii::log('returning '.print_r($criteria,true), 'debug', 'AllTicketsModel::getSearch()');
 153          return $criteria;
 154          }

 156      /**
 157       * returns fromatted ticket object data as an array, suitable for base class jqGrid-data handling
 158       * 
 159       * @param  CRSTicketAppService_ticketVO $t Ticket object
 160       * @return array  array-format fromatted ticket data
 161       * @access public
 162       */
 163  	public function addResults($t)
 164          {
 165          $d=new DateTime($t->creationDate);
 166          $d->add(Yii::app()->params['ticketExpiryDays']);
 167          $ticket_expiry_date = $d->format('Y-m-d');
 168          $ticket_holder = '?';
 169          if(property_exists($t->operation,'domainHolderField') and property_exists($t->operation->domainHolderField,'newValue'))
 170              $ticket_holder = $t->operation->domainHolderField->newValue;
 171          return array
 172              (
 173              'A'=>$t->id,
 174              'B'=>$t->domainName,
 175              'C'=>$ticket_holder,
 176              'D'=>decipher_admin_status($t->adminStatus),
 177              'E'=>decipher_tech_status($t->techStatus),
 178              'F'=>$t->type,
 179              'G'=>$ticket_expiry_date,
 180              );
 181          }
