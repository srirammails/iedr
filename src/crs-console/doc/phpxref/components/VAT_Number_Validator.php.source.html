  29  class VAT_Number_Validator
  30      extends RegexValidator
  31      {
  32  
  33      /**
  34       * constructor, sets up regex string
  35       * 
  36       * Regex derived from info on pages 278 & 279 of {@link http://my.safaribooksonline.com/book/programming/regular-expressions/9780596802837/validation-and-formatting/284#X2ludGVybmFsX0ZsYXNoUmVhZGVyP3htbGlkPTk3ODA1OTY4MDI4MzcvMjc4}
  37       * Regexes update from {@link http://www.iecomputersystems.ie/ordering/eu_vat_numbers.htm}
  38       * See also {@link http://ec.europa.eu/taxation_customs/vies/faqvies.do#item11}
  39       * 
  40       * @return void  
  41       * @access public
  42       */
  43  	public function __construct()
  44          {
  45          $EU_Vat_Regex = '^('.
  46              # Austria
  47              '(AT)U[0-9]{8}|'.
  48              # Belgium
  49              '(BE)[0-9]{9}|'.
  50              # Bulgaria
  51              '(BG)[0-9]{9,10}|'.
  52              # Cyprus
  53              '(CY)[0-9]{8,9,10}|'.
  54              # Czech Republic
  55              '(CZ)[0-9]{8,9,10}|'.
  56              # Denmark
  57              '(DK)[0-9]{8}|'.
  58              # Estonia
  59              '(EE)[0-9]{9}|'.
  60              # Finland
  61              '(FI)[0-9]{8}|'.
  62              # France (dont use I or O)
  63              '(FR)[0-9A-HJ-NP-Z]{2}[0-9]{9}|'.
  64              # Germany
  65              '(DE)[0-9]{9}|'.
  66              # Greece
  67              '(EL|GR)0[0-9]{8}|'.
  68              # Hungary
  69              '(HU)[0-9]{8}|'.
  70              # Ireland
  71              '(IE)?[0-9][0-9A-Z+*][0-9]{5}[A-Z]|'.
  72              # Italy
  73              '(IT)[0-9]{11}|'.
  74              # Latvia
  75              '(LV)[0-9]{11}|'.
  76              # Lithuania
  77              '(LT)([0-9]{9}|[0-9]{12})|'.
  78              # Luxembourg
  79              '(LU)[0-9]{8}|'.
  80              # Malta
  81              '(MT)[0-9]{8}|'.
  82              # Netherlands
  83              '(NL)[0-9]{9}B[0-9]{2}|'.
  84              # Poland
  85              '(PL)[0-9]{10}|'.
  86              # Portugal
  87              '(PT)[0-9]{9}|'.
  88              # Romania
  89              '(RO)[0-9]{2,10}|'.
  90              # Slovakia
  91              '(SK)[0-9]{10}|'.
  92              # Slovenia
  93              '(SI)[0-9]{8}|'.
  94              # Spain
  95              '(ES)([A-Z][0-9]{8}|[A-Z][0-9]{7}[A-Z]|[0-9]{8}[A-Z])|'.
  96              # Sweden
  97              '(SE)[0-9]{10}01|'.
  98              # United Kingdom
  99              '(GB)([0-9]{9}([0-9]{3})?|[A-Z]{2}[0-9]{3})'.
 100              ')$';
 101  
 102          $this->regex = '/'.$EU_Vat_Regex.'/';
 103          $this->invalid_msg = 'is not a <a href="http://ec.europa.eu/taxation_customs/vies/faqvies.do#item11">valid EU VAT Number</a>';
 104          }

 106      /**
 107       * trims away non-alphanumeric characters and validates, setting model error message if invalid
 108       * 
 109       * @todo   could be refactored to make better use of base class functionality
 110       * @param  CModel     $model     model with data to validate
 111       * @param  string     $attribute name of model attribute to validate
 112       * @return boolean    true if valid
 113       * @access protected
 114       */
 115  	protected function validateAttribute($model,$attribute)
 116          {
 117          if($model && $attribute)
 118              {
 119              $vat = ''.$model->$attribute;
 120              $trimmed = preg_replace('/[^A-Z0-9+*]/i','',$vat);
 121              if(strlen($vat)==0 or preg_match($this->regex, $trimmed)===1)
 122                  return true;
 123                  // maybe also validate/lookup online : http://ec.europa.eu/taxation_customs/vies/checkVatService.wsdl
 124              }
 125          $lbls = $model->attributeLabels();
 126          $lbl = $lbls[$attribute];
 127          $model->addError($attribute,Yii::t('yii',$lbl.' '.$this->invalid_msg));
 128          return false;
 129          }
 130  
 131      }
