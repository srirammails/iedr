  94       * returns Excel file

  96       * @return string binary excel file data
  97       * @access public
  98       */
  99  	public function output() {
 100          $out = $this->xlsBOF();
 101          $row = 0; $col = 0;
 102          foreach($this->colheads as $hdrdata) {
 103              $out.=$this->xlsWriteLabel($row,$col++,$hdrdata);
 104              }
 105          foreach($this->data as $rowarr) {
 106              ++$row; $col=0;
 107              foreach($rowarr as $celldata) {
 108                  $out.=$this->xlsWriteLabel($row,$col++,$celldata);
 109                  }
 110              }
 111          $out.=$this->xlsEOF();
 112          return $out;
 113          }

 121  	protected function xlsBOF() {
 122          return pack('ssssss', 0x809, 0x8, 0x0, 0x10, 0x0, 0x0);

 135      /**
 136       * returns BIFF5 formatted number
 137       * 
 138       * @param  integer   $Row   row number

 144  	protected function xlsWriteNumber($Row, $Col, $Value) {
 145          return pack('sssss', 0x203, 14, $Row, $Col, 0x0).pack('d', $Value);
 146          }

 148      /**
 149       * returns BIFF5 formatted string
 150       * 
 151       * @param  integer   $Row   row number
 152       * @param  integer   $Col   column number
 153       * @param  string    $Value cell value
 154       * @return string    BIFF5 formatted string
 155       * @access protected
 156       */
 157  	protected function xlsWriteLabel($Row, $Col, $Value ) {
 158          $L = strlen($Value);
 159          return pack('ssssss', 0x204, 8 + $L, $Row, $Col, 0x0, $L).$Value;
 160          } 
 161      }
