 100              array('creditcardno,euros_amount,exp_year,exp_month',

 102                  'on'=>'topup,'.CreditCardFormSegmentModel::$without_amount_scenario),
 103              array('exp_year',
 104                  'numerical', 'min'=>$year, 'max'=>($year+15),
 105                  'on'=>'topup,'.CreditCardFormSegmentModel::$without_amount_scenario),
 106              array('exp_month',
 107                  'numerical', 'min'=>1, 'max'=>12,
 108                  'on'=>'topup,'.CreditCardFormSegmentModel::$without_amount_scenario),
 109              array('cvn',
 110                  'numerical', 'min'=>0, 'max'=>9999,
 111                  'on'=>'topup,'.CreditCardFormSegmentModel::$without_amount_scenario),
 112              array('creditcardno',
 113                  'CreditCardValidator',
 114                  'on'=>'topup,'.CreditCardFormSegmentModel::$without_amount_scenario),

 146      /**
 147       * returns an object instance of class CRSPaymentAppService_paymentRequestVO, filled with the model's credit-card data
 148       * 
 149       * @param  DynamicModelBase $mdl model whose data will be copied
 150       * @return CRSPaymentAppService_paymentRequestVO object filled with the model's credit-card data
 151       * @access public
 152       * @static
 153       */
 154  	static public function asWSAPIPaymentObject($mdl)
 155          {
 156          // call with a parameter being either an instance of this class or a class that has it's values mixed in
 157          // e.g. $p = CreditCardFormSegmentModel::asWSAPIPaymentObject($m);
 158          $payreq = new CRSPaymentAppService_paymentRequestVO();
 159          $payreq->currency       = 'EUR';
 160          $payreq->amount        = $mdl->euros_amount * 100;
 161          $payreq->cardNumber     = $mdl->creditcardno;
 162          $payreq->cardExpDate    = sprintf('%02u%02u',$mdl->exp_month,$mdl->exp_year);
 163          $payreq->cardHolderName    = $mdl->holder;
 164  
 165          // determine the card type
 166          CreditCardValidator::CCPrefixLengthTypeCheck($mdl->creditcardno,$payreq->cardType);
 167          switch($payreq->cardType)
 168              {
 169              case CREDCARDTYPE_VISA:
 170              case CREDCARDTYPE_MASTERCARD:
 171                  $payreq->cvnPresenceIndicator = CVN_PRESENT_OK;
 172                  break;
 173              case CREDCARDTYPE_LASER:
 174                  $payreq->cvnPresenceIndicator = CVN_PRESENT_NOT_ON_CARD;
 175                  break;
 176              }
 177          $payreq->cvnNumber = $mdl->cvn;
 178  
 179          return $payreq;
 180          }
