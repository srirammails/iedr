 109      /**
 110       * Attempt authentication
 111       * 
 112       * clears Yii Application Identity, and all user-related data, then attempts to authenticate via the backend CRS-WS API.
 113       * 
 114       * @return boolean true if authentication succeeded
 115       * @access public 
 116       */
 117  	public function authenticate()
 118          {
 119          $this->errorCode = self::ERROR_USERNAME_INVALID;
 120          $this->_id = null;
 121          Yii::app()->user->setId(null);
 122          $this->clearUserData();
 123          $this->setState(AUTHENTICATED, false);
 124  
 125          if(!(isset($this->username) and $this->username!='' and strlen($this->username)>0 and isset($this->password) and $this->password!='' and strlen($this->password)>0))
 126              {
 127              Yii::log('Empty Credentials : '.print_r($this,true), 'error', 'NicHandleIdentity::authenticate()');
 128              exit;
 129              }
 130  
 131          if($this->CRSAuthenticationFunction($this->username, $this->password))
 132              {
 133              $this->errorCode=self::ERROR_NONE;
 134              $this->setState(AUTHENTICATED, true);
 135              }
 136          else
 137              $this->errorCode=self::ERROR_UNKNOWN_IDENTITY;
 138          return ($this->errorCode==self::ERROR_NONE);
 139          }

 160      /**
 161       * switch to another registrar's account
 162       * 
 163       * This function is called as an alternative to authentication, in the special case of user-switching.
 164       * 
 165       * The logged-in user, as stored in Yii::app()->user->actualInternalUser, is stored locally,
 166       * then the normal authentication process sets up the NicHandleIdentity as if the switch-to user
 167       * had authenticated successfully.
 168       * After this data setup, the Yii::app()->user->actualInternalUser is restored to it's previous
 169       * value.
 170       * 
 171       * @todo As of "CRS version r2746 (2011-06-21 11:54)", Nask have updated their Authentication API
 172       *       to use a real cryptographic hash - instead of simply a copy of the username - as the API
 173       *       authentication token, so the current strategy (this function) will no longer work.
 174       *       What needs to be implemented instead a new CRS API that performs a "login as" functionality
 175       *       using username, password, and login-as-username.
 176       *       After NASK implement that, this class will need to be somewhat rewritten.
 177       *
 178       * @return void  
 179       * @access public
 180       */
 181  	public function switchToAccount()
 182          {
 183          // create new auth object
 184          $auvo = new CRSDomainAppService_authenticatedUserVO();
 185          $auvo->username = $this->getName();
 186          $auvo->authenticationToken = $this->getName(); // BUG will have to change this with some new Nask API to get a real Auth token without that account's password
 187          // set app identity to this new one : same as login, but skip authentication
 188          $actual = Yii::app()->user->actualInternalUser;
 189          $this->clearUserData();
 190          $this->setUserData($auvo);
 191          $this->setState(ACTUAL_INTERNAL_USER,$actual);
 192          Yii::log('User Switched to '.$auvo->username, 'debug', 'NicHandleIdentity::switchToAccount()');
 193          }

 196       * Authenticate login credentials via the CRS-WS API.
 197       * 
 198       * On authentication success, loads user-specific data
 199       * 
 200       * @param  unknown   $nichandle username (nic-handle)
 201       * @param  unknown   $nicpass   password
 202       * @return boolean   true if successful
 203       * @access protected
 204       */
 205  	protected function CRSAuthenticationFunction($nichandle, $nicpass)
 206          {
 207          $auvo = null; // CRSDomainAppService_authenticatedUserVO
 208          if(CRSAuthenticationService_service::authenticate($auvo, $this->errorMessage, $nichandle, $nicpass))
 209              {
 210              $this->setState(ACTUAL_INTERNAL_USER, $nichandle);
 211              $this->setUserData($auvo);
 212              return true;
 213              }
 214          return false;
 215          }

 247       * @access public

 249  	public function setUserData($auvo)
 250          {
 251          // logged in OK
 252          $this->setState(AUTHENTICATED_USER_VO, $auvo);
 253          $nicdetails = $this->getNICDetails($auvo);
 254          if($nicdetails!=null) {
 255              $this->_id = $nicdetails->accountId;
 256              $this->setState(NIC_DETAILS, $nicdetails);
 257              }
 258          $this->setState(RESELLER_DEFAULTS, $this->getResellerDefaults($auvo));
 259          $this->setState(AUTH_GROUPS, $this->getAuthGroups($auvo));
 260          $this->getDepositAccountLimits($auvo);
 261          $this->getResellerAccountDetails($this->_id);
 262          $linvd = $this->getLastInvoiceDate($auvo);
 263          $this->setState(LAST_INVOICE_DATE, $linvd);
 264  
 265          Yii::log('set (LAST_INVOICE_DATE) '.LAST_INVOICE_DATE.'=', 'debug', 'NicHandleIdentity::getNICDetails()');

 303       * returns an array of authorization groups that the current logged-in user belongs to, as per CRS-WS API
 304       * 
 305       * @param  authenticatedUserVO    $auvo authentication token
 306       * @return array     array of authorization groups

 309  	protected function getAuthGroups($auvo)
 310          {
 311          crs_wsapi_service_base::$log_quench = true;
 312          $authgroups = array();
 313          $grouplist = array('Guest','Registrar','CustomerService','Finance','Hostmaster','HostmasterLead','Batch','TechnicalLead','Technical','FinanceLead','SuperRegistrar',);
 314          foreach($grouplist as $level)
 315              {
 316              $result = null;
 317              $errs   = null;
 318              CRSPermissionsAppService_service::isUserInGroup($result,$errs,$auvo, $level);
 319              if($result==1)
 320                  $authgroups[] = $level;
 321              }
 322          crs_wsapi_service_base::$log_quench = false;
 323          Yii::log(print_r($authgroups,true), 'debug', 'NicHandleIdentity::getAuthGroups()');
 324          return $authgroups;
 325          }

 356          {
 357          $sup_auvo = null; // CRSDomainAppService_authenticatedUserVO
 358          $timeout = 3600; // TODO set to max session lifetime, once NASK have a session timeout
 359          $cache_key = 'superuser_auvo';
 360          $errs = null;
 361          $sup_auvo = Yii::app()->cache->get($cache_key);
 362          if($sup_auvo==null)
 363              {
 364              $sup_nic = Yii::app()->params['superuser_nic'];
 365              $sup_pass = Yii::app()->params['superuser_pass'];
 366              if(!CRSAuthenticationService_service::authenticate($sup_auvo, $errs, $sup_nic, $sup_pass))

 372      /**
 373       * gets, via CRS-WS API, some account details for given account ID, with 'superuser' authentication token.
 374       * 
 375       * Details fetched : 
 376       * - tariff: one of { Deleted,RetailB,Silver,Trade,TradeA,TradeAChar,TradeB,TradeBChar }
 377       * - agreementSigned: 0 or 1
 378       * - ticketEdit: 0 or 1

 382       * @access protected

 384  	protected function getResellerAccountDetails($accid)
 385          {
 386          $result = null;
 387          $errs   = null;
 388          $suvo = NicHandleIdentity::getSuperUserAuthentication();
 389          if($suvo!=null)
 390              {
 391              CRSResellerAppService_service::get($result,$errs, $suvo, $accid);
 392              if($errs==null)
 393                  {
 394                  Yii::log(print_r($result,true), 'debug', 'NicHandleIdentity::getResellerAccount()');
 395                  $this->setState(TARIFF, $result->tariff);
 396                  $this->setState(AGREEMENTSIGNED, $result->agreementSigned);
 397                  $this->setState(TICKETEDIT, $result->ticketEdit);
