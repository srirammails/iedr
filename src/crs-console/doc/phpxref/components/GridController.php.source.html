  99      /**
 100       * Shared common functionality for jqGrid display
 101       * 
 102       * Common functionality for jqGrid display, including CSV and Excel data export,
 103       * JSON encoding of grid data for ajax loading, pagination, and handling of searching and sorting parameters
 104       * 
 105       * @return void      
 106       * @access public    
 107       * @throws CException thrown if function called with @GridModelBase lacking a default sort column
 108       * @see    GridSelectionModel
 109       */
 110  	public function actionGetgriddata()
 111          {
 112          $model = self::getModelFromQuery();
 113          if(!isset($model->defaultSortColumn))
 114              throw new CException('Model must have defaultSortColumn set to a model-column-name with unique value');
 115          if(is_object($model) and isset($_GET['oper']))
 116              {
 117              Yii::log('oper='.$_GET['oper'], 'debug', 'GridController::actionGetgriddata():'.__LINE__);
 118              switch($_GET['oper'])
 119                  {
 120                  case 'csv': case 'excel':
 121                      $data = $this->getDataReq($model,$_GET);
 122                      $heads = array();
 123                      foreach($model->columns as $col)
 124                          $heads[] = $col['label'];
 125                      $p = null;
 126                      $filename = $this->getExportFilenameTag().Yii::app()->user->name.'-'.date('Y-m-d');
 127                      switch($_GET['oper'])
 128                          {
 129                          case 'csv'  : $p = new CreateCSVFile        ($filename,$heads,$data['rows']); break;
 130                          case 'excel': $p = new CreateExcelSpreadsheet    ($filename,$heads,$data['rows']); break;
 131                          }
 132                      if($p != null) {
 133                          $p->headers();
 134                          echo $p->output();
 135                          }
 136                      break;
 137                  // case 'pdf': some other time :)
 138                  case 'grid':
 139                      $data = $this->getDataReq($model,$_GET,/*paging=*/true);
 140                      $modelColIDs = array_keys($model->columns);
 141                      foreach($data['rows'] as $k => $r)
 142                          {
 143                          $rowArray = array();
 144                          $rowArray['id'] = $r[$model->defaultSortColumn];
 145                          $rowArray['cell'] = array();
 146                          foreach($modelColIDs as $colID)
 147                              array_push($rowArray['cell'],$r[$colID]);
 148                          $data['rows'][$k] = $rowArray;
 149                          }
 150                      header('Content-Type: text/x-json;charset=utf-8');
 151                      echo json_encode($data);
 152                      break;
 153                  default:
 154                      echo 'Unrecognised "oper"';
 155                      break;
 156                  }
 157              Yii::app()->end();
 158              }
 159          else
 160              {
 161              Yii::log('ERR, request='.print_r($_REQUEST,true), 'debug', 'GridController::actionGetgriddata():'.__LINE__);
 162              $this->render('site/error', 'invalid request');
 163              }
 164          }

 276      /**
 277       * Extract search and sort parameters from request GET parameters; return Array.
 278       * 
 279       * Called from actionGetgriddata() when fetching JSON grid data.
 280       * Extracts jqGrid url parameters and forwards them to model functions to produce criteria for data retieval
 281       * 
 282       * @param  mixed     $model  grid model instance, which further processes the search/sort parameters
 283       * @param  array     $params http request parameters array
 284       * @param  boolean   $paging whether pagination parameters are needed
 285       * @return mixed     data array returned by call to @getData
 286       * @access protected
 287       */
 288  	protected function getDataReq($model,$params,$paging=false)
 289          {
 290          Yii::log('ENTER', 'debug', 'GridController::getDataReq()');
 291          $perpage     = $paging ? $params['rows'] : 999999999999;
 292          $pagenum     = $paging ? $params['page'] : 1; // page is 1-based
 293          $sortparams    = array('col'=>$params['sidx'],'ord'=>$params['sord']);
 294          $searchparams     = array();
 295          if(isset($params['_search']) and $params['_search']=='true')
 296              {
 297              // TODO: handle date filter as a date range
 298              if(isset($params['filters']))
 299                  {
 300                  $filters = $params['filters'];
 301                  if(strlen($filters)>0 and strpos($filters,'groupOp')>=0)
 302                      {
 303                      $filter = json_decode(urldecode($filters));
 304                      if(is_object($filter) and $filter->rules)
 305                          $searchparams = $filter->rules;
 306                      }
 307                  }
 308              else
 309                  if(isset($params['searchField']) and isset($params['searchString']))
 310                      {
 311                      $o = new stdClass();
 312                      $o->field = $params['searchField'];
 313                      $o->data  = $params['searchString'];
 314                      $searchparams[0] = $o;
 315                      }
 316                  else
 317                      {
 318                      // new jqGrid4 search mode, where a col-id contains a filter value
 319                      // iterate over model column names, check if there's a filter value
 320                      foreach($model->columns as $idx => $col)
 321                          {
 322                          if(isset($params[$idx]) and strlen($params[$idx])>1)
 323                              {
 324                              $o = new stdClass();
 325                              $o->field = $idx; // 'field' ?
 326                              $o->data  = $params[$idx]; // search string
 327                              $searchparams[] = $o;
 328                              }
 329                          }
 330                      }
 331              }
 332          $criteria     = $model->getSearch($searchparams);
 333          $sort         = $model->getSort($sortparams);
 334          return $this->getData($model,$criteria,$sort,$perpage,$pagenum);
 335          }

 337      /**
 338       * Runs backend query for data with the supplied parameters, returns an array
 339       * 
 340       * @param  object    $model    model instance, to create row-array from returned backend data
 341       * @param  CRSDomainAppService_domainSearchCriteria    $criteria search criteria
 342       * @param  CRSDomainAppService_sortCriterion    $sort     sort criteria
 343       * @param  integer   $perpage  number of rows per page wanted
 344       * @param  integer   $pagenum  which page of paginated data, used to calculate row offset
 345       * @return array     array of data, structured like : array('records'=>X,'page'=>Y,'total'=>Z,'rows'=>array( ... ));
 346       * @access protected
 347       */
 348  	protected function getData($model,$criteria,$sort,$perpage,$pagenum)
 349          {
 350          // Get Grid data with search and sort parameters; return Array.
 351          // Called from actionGetgriddata()->getDataReq() when fetching JSON grid data
 352          Yii::log('ENTER', 'debug', 'GridController::getData()');
 353          // paging
 354          $pagedata    = array('records'=>0,'page'=>$pagenum,'total'=>0,'rows'=>array());
 355          $offset     = $perpage * ($pagenum-1);
 356          $limit      = $perpage;
 357          // API SOAP call
 358          $result     = new CRSDomainAppService_domainSearchResultVO();
 359          $errs         = null;
 360          // BUG TODO check that the criteria mention the correct unambiguous column name, e.g. : "D.D_Name" instead of "D_Name"
 361          /* BUG HACK workaround NASK not checking input for NULLs */ if($criteria->contactType == null) unset($criteria->contactType);
 362          $user = Yii::app()->user->authenticatedUser;
 363          CRSDomainAppService_service::findDomains($result,$errs, $user, $criteria, $offset, $limit, $sort);
 364          if(count($errs)==0)
 365              {
 366              // records = total records on all pages
 367              $pagedata['records']     = $result->totalResults;
 368              // total = total num result pages = (t div p)+1
 369              $pagedata['total']     = math_div($result->totalResults,$perpage)+1;
 370              if(isset($result->results))
 371                  {
 372                  if(is_array($result->results))
 373                      {
 374                      foreach($result->results as $k => $o)
 375                          $pagedata['rows'][] = $model->addResults($o);
 376                      }
 377                  else if(is_object($result->results))
 378                      {
 379                      $pagedata['rows'][] = $model->addResults($result->results);
 380                      }
 381                  else Yii::log(print_r($result,true), 'warning', 'GridController::getData() : results format?'.__LINE__);
 382                  }
 383              else Yii::log(print_r($result,true), 'warning', 'GridController::getData() : WARN : no \'results\':'.__LINE__);
 384              }
 385          else Yii::log(print_r($errs,true), 'error', 'GridController::getData() : ERROR:'.__LINE__);
 386          return $pagedata;
 387          }
