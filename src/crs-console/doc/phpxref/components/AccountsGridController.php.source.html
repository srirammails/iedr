 100  	protected function execCommand($gsc_model)
 101          {
 102          // called by GridController::actionConfirmaction()
 103          $this->invoiceType = $gsc_model->invoiceType;
 104          $gsc_model->commandresults = array();
 105          $invoiceType = $this->getNeededInvoiceType();
 106          $gsc_model->euros_amount = 0.0;
 107          $pRequest = ($gsc_model->needCreditCard) ? CreditCardFormSegmentModel::asWSAPIPaymentObject($gsc_model) : null;
 108          $isTest = '0';
 109          $remoteAddr = $_SERVER['REMOTE_ADDR'];
 110          $totalFee = 0.0;
 111          $totalVat = 0.0;
 112          $this->backend_errors = array();
 113  
 114          if($gsc_model->command == 'msd')
 115              {
 116              $this->doToMSD($gsc_model);
 117              }
 118          else
 119              {
 120              switch($gsc_model->command)
 121                  {
 122                  case 'dep':
 123                      // prepare domain list as Array of CRSPaymentAppService_domainWithPeriodVO
 124                      $dwpArr = array();
 125                      $lastPeriod = 0;
 126                      foreach($this->groupCommandByPeriod($gsc_model) as $period => $domainlist)
 127                          {
 128                          foreach($domainlist as $d)
 129                              {
 130                              $dwp = new CRSPaymentAppService_domainWithPeriodVO();
 131                              $dwp->domainName = $d;
 132                              $dwp->period     = $period;
 133                              $dwpArr[] = $dwp;
 134                              }
 135                          }
 136                      CRSPaymentAppService_service::payFromDeposit($result,$errs,Yii::app()->user->authenticatedUser,$remoteAddr,$dwpArr,$invoiceType,$isTest);
 137                      if($result!=null and is_object($result))
 138                          {
 139                          $totalFee += $result->amount;
 140                          $totalVat += $result->vat;
 141                          foreach($this->groupCommandByPeriod($gsc_model) as $period => $domainlist)
 142                              {
 143                              // the results are for the whole collection of periods and domains, but we display results PER-Period
 144                              $result->amount = null;
 145                              $result->vat    = null;
 146                              $gsc_model->commandresults[$period] = array('domainlist'=>$domainlist,'results'=>$result,'errs'=>null);
 147                              }
 148                          }
 149                      else
 150                          $this->backend_errors[] = $errs;
 151                      break;
 152                  default:
 153                      foreach($this->groupCommandByPeriod($gsc_model) as $period => $domainlist)
 154                          {
 155                          $result = null;
 156                          switch($gsc_model->command)
 157                              {
 158                              case 'on':
 159                                  CRSPaymentAppService_service::payOnline($result,$errs,Yii::app()->user->authenticatedUser,$pRequest,$remoteAddr,$domainlist,$invoiceType,$isTest);
 160                                  break;
 161                              case 'off':
 162                                  CRSPaymentAppService_service::payOffline($result,$errs,Yii::app()->user->authenticatedUser,$remoteAddr,$domainlist,$invoiceType,$isTest);
 163                                  break;
 164                              case 'dep': /* parameter is now array of CRSPaymentAppService_domainWithPeriodVO objects */ break;
 165                              case 'payonline': case 'paydeposit': case 'reactivate':
 166                                  CRSPaymentAppService_service::payMsdReactivation($result,$errs,Yii::app()->user->authenticatedUser,$pRequest,$remoteAddr,$domainlist,$gsc_model->needCreditCard,$isTest);
 167                                  break;
 168                              }
 169                          if($result!=null and is_object($result))
 170                              {
 171                              $totalFee += $result->amount;
 172                              $totalVat += $result->vat;
 173                              }
 174                          else
 175                              $this->backend_errors[$period] = $errs;
 176                          $gsc_model->commandresults[$period] = array('domainlist'=>$domainlist,'results'=>$result,'errs'=>$errs);
 177                          }
 178                      break;
 179                  }
 180  
 181              }
 182          // force re-fetch of deposit balance
 183          invalidate_DepositBalance(Yii::app()->user->authenticatedUser->username);
 184          $gsc_model->commandresults['TOTAL'] = array('totalFee'=>$totalFee,'totalVat'=>$totalVat);
 185          Yii::log('command "'.$gsc_model->command.'" done, backend_errors='.print_r($this->backend_errors,true), 'debug', 'AccountsGridController::actionConfirmaction()');
 186          }

 222       * returns data from the gridSelection model grouped by renewal period
 223       * 
 224       * periods are derived from product codes with an array lookup via {@link get_reg_prices}
 225       * 
 226       * @param  mixed     $gsc_model gridSelection model
 227       * @return array     an array of renewal-periods with subarrays of sets of selected domains with the same period
 228       * @access protected
 229       */
 230  	protected function groupCommandByPeriod($gsc_model)
 231          {
 232          #Yii::log(print_r($gsc_model,true), 'debug', 'AccountsGridController::groupCommandByPeriod()');
 233          $regarr = get_reg_prices();
 234          $group_period_domain = array();
 235          foreach($gsc_model->domainListToArray() as $dom)
 236              {
 237              $dom_prefix = DynamicDomainListModel::domainToVarPrefix($dom);
 238              $dom_prod_name = $dom_prefix.'_prodcode';
 239              $dom_prodcode = $gsc_model->$dom_prod_name;
 240              // use period rather than product-code
 241              $prodObj = $regarr['CODE'][$dom_prodcode];
 242              $group_period_domain[$prodObj->duration][] = $dom;
 243              }
 244          Yii::log('groupCommandByPeriod => '.print_r($group_period_domain,true), 'debug', 'AccountsGridController::groupCommandByPeriod()');
 245          return $group_period_domain;

 276      /**
 277       * Get Invoice summary for a model to find the count of records available for given criteria
 278       * 
 279       * Calls backend Soap function depending on invoice type.
 280       * This function is needed simply because NASK implemented the getInvoices APIs without
 281       * record-counts in the returned data structures, unlike the findDomain API functions.
 282       * 
 283       * @param  CRSPaymentAppService_invoiceType  $invoiceType type of invoice
 284       * @param  string $month       year and month in format 'YYYY-MM'
 285       * @param  string $domainName  optional domain name
 286       * @return object  object with domain count for invoice
 287       * @access public 
 288       */
 289  	public function getInvoicesSummary($invoiceType=null,$month=null,$domainName=null)
 290          {
 291          if($invoiceType==null) $invoiceType = $this->getNeededInvoiceType();
 292          $result = null;
 293          /*
 294          return types:
 295              CRSPaymentAppService_shortInvoiceVO
 296                  public $invoiceNo;// type=xs:string, (NOT NULL), min=0
 297                  public $vat;// type=xs:decimal, (NOT NULL), min=0
 298                  public $amount;// type=xs:decimal, (NOT NULL), min=0
 299                  public $count;// type=xs:int, (NOT NULL), min=1
 300              CRSPaymentAppService_invoicesSummaryVO
 301                  public $count;// type=xs:int, (NOT NULL), min=1
 302                  public $amountSum;// type=xs:decimal, (NOT NULL), min=0
 303                  public $vatSum;// type=xs:decimal, (NOT NULL), min=0
 304          */
 305          switch($invoiceType)
 306              {
 307              case CRSPaymentAppService_invoiceType::_MSD:
 308                  $renewalDateType = 'CURRENT';
 309                  $noPendingPayment = true;
 310                  CRSPaymentAppService_service::getMsdInvoiceCount($result,$this->backend_errors, Yii::app()->user->authenticatedUser, $noPendingPayment, $renewalDateType);
 311                  // returns integer
 312                  Yii::log('getMsdInvoiceCount='.print_r($result,true), 'debug', 'AccountsGridController::getInvoicesSummary()'.__LINE__);
 313                  break;
 314              case CRSPaymentAppService_invoiceType::_TRAN:
 315                  CRSPaymentAppService_service::getTransferShortInvoice($result,$this->backend_errors, Yii::app()->user->authenticatedUser);
 316                  // returns CRSPaymentAppService_shortInvoiceVO
 317                  Yii::log('getTransferShortInvoice='.print_r($result,true), 'debug', 'AccountsGridController::getInvoicesSummary()'.__LINE__);
 318                  break;
 319              case CRSPaymentAppService_invoiceType::_TRAN_ADV:
 320              case CRSPaymentAppService_invoiceType::_FUT_REN:
 321                  CRSPaymentAppService_service::getInvoicesSummary($result,$this->backend_errors, Yii::app()->user->authenticatedUser, $invoiceType, $domainName, $month);
 322                  // returns CRSPaymentAppService_invoicesSummaryVO
 323                  Yii::log('getInvoicesSummary='.print_r($result,true), 'debug', 'AccountsGridController::getInvoicesSummary()'.__LINE__);
 324                  break;
 325              case CRSPaymentAppService_invoiceType::_CURR_REN_REG:
 326                  CRSPaymentAppService_service::getCurRegRenShortInvoice($result,$this->backend_errors, Yii::app()->user->authenticatedUser);
 327                  // returns CRSPaymentAppService_shortInvoiceVO
 328                  Yii::log('getCurRegRenShortInvoice='.print_r($result,true), 'debug', 'AccountsGridController::getInvoicesSummary()'.__LINE__);
 329                  $result->invoiceNumber = $result->invoiceNo;
 330                  break;
 331              }
 332          if(count($this->backend_errors)==0)
 333              {
 334              Yii::log('InvoiceType='.$invoiceType.', getInvoicesSummary='.print_r($result,true), 'debug', 'AccountsGridController::getInvoicesSummary()'.__LINE__);
 335              switch($invoiceType)
 336                  {
 337                  case CRSPaymentAppService_invoiceType::_MSD:
 338                      $num = $result;
 339                      $result = new stdClass();
 340                      $result->count = $num;
 341                      $result->invoiceNumber = 'N/A';
 342                      break;
 343                  case CRSPaymentAppService_invoiceType::_TRAN:
 344                  case CRSPaymentAppService_invoiceType::_TRAN_ADV:
 345                  case CRSPaymentAppService_invoiceType::_FUT_REN:
 346                  case CRSPaymentAppService_invoiceType::_CURR_REN_REG:
 347                  default:
 348                      break;
 349                  }
 350              }
 351          else Yii::log(print_r($this->backend_errors,true), 'error', 'AccountsGridController::getInvoicesSummary()'.__LINE__);
 352          return $result;
 353          }
