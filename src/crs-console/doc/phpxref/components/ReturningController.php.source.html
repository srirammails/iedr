  14  /**
  15   * base class containing functions to output returning-form html
  16   * 
  17   * Returning forms are forms which, when sucessfully submitted, offer the option of returning to a previous page,
  18   * then setting some value on the previous page to a value which was found or created on the returning-form.
  19   * This pattern is useful for e.g. digressing from the domain-creation form to create a nic-handle, then
  20   * return to the domain-creation form, filling in some field with the newly-created nic-handle.
  21   * Finding nic-handles is another use of returning forms.
  22   * 
  23   * @category  NRC
  24   * @package   IEDR_New_Registrars_Console
  25   * @author    IEDR <asd@iedr.ie>
  26   * @copyright 2011 IEDR
  27   * @license   http://www.iedr.ie/ (C) IEDR 2011
  28   * @version   Release: @package_version@
  29   * @link      https://statistics.iedr.ie/
  30   * @see       References to other sections (if any)...
  31   */
  32  class ReturningController
  33      extends AuthOnlyBaseController
  34      {
  35  
  36      /**
  37       * outputs javascript for handling a selected row in a jqGrid on the same page.
  38       * 
  39       * intended for use when finding a record, whose ID we want to return to the original form.
  40       * 
  41       * @param  unknown $selectionID html unique class ID of the selected jqGrid's cell containing the primary-key value of the selected Row,
  42       * @param  unknown $buttonLabel unique html button label, to identify submit-button to un-disable
  43       * @return void   
  44       * @access public 
  45       */
  46  	public function outputReturningFormJS($selectionID,$buttonLabel)
  47          {
  48  echo <<<EOF1
  49  <script>
  50  function handleSelectedRow(z) {
  51      var oldval = jQuery('#$selectionID').val();
  52      alert("ReturningController-generated stuff (old="+oldval+") "+z+" into id $selectionID");
  53      // [1] set selected row-value into selectionID
  54      jQuery('#$selectionID').val(z);
  55      // [2] enable form-post button
  56      jQuery('[value="$buttonLabel"]').disabled = false;
  57      }
  58  </script>
  59  EOF1;
  60          }
  61  
  62      /**
  63       * outputs form html and button for returning to a previous page with a found value
  64       * 
  65       * @param  object  $model       the model containing any data to be rendered by the calling view
  66       * @param  string  $selectionID html tag ID of hidden form field to hold returned value
  67       * @param  string  $buttonLabel text for return button
  68       * @return void   
  69       * @access public 
  70       */
  71  	public function outputReturningFormHtml($model,$selectionID,$buttonLabel='Return to Form')
  72          {
  73          if(is_array($model->returningData) and isset($model->returningData['returnurl']) and
  74          $model->returningData['returnurl'] and strlen($model->returningData['returnurl'])>0)
  75              {
  76              // output javascript to catch selected Row, to copy data to it's hidden form
  77              $this->outputReturningFormJS($selectionID,$buttonLabel);
  78              // output returning-form ; called from View to render model data ; posts back to the calling form
  79              $form = $this->beginWidget('CActiveForm',array('action'=>$model->returningData['returnurl']));
  80              foreach($model->getReturningDataNames() as $v)
  81                  echo $form->hiddenField($model,'returningData['.$v.']');
  82              echo CHtml::submitButton($buttonLabel/*,array('disabled'=>'disabled')*/);
  83              $this->endWidget();
  84              }
  85          #else Yii::log('NOT Outputting ReturningForm :: '.print_r($this,true)."\n".print_r($model,true), 'error', 'ReturningController::outputReturningFormHtml()');
  86          // no output if there is no returning data
  87          }
  88      }
