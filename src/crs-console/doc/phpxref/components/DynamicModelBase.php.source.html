  85      /**
  86       * Adds another model's attributes, validation rules, and field labels to this class
  87       * 
  88       * @param  {@link FormModelBase}     $model other model to add
  89       * @return void      
  90       * @access public    
  91       * @throws CException thrown if model parameter is null or not a descendant of {@link FormModelBase}
  92       */
  93  	public function addMixinModel($model)
  94          {
  95          // copy member variables
  96          if(($model!=null) and ($model instanceof FormModelBase))
  97              {
  98              foreach(get_object_vars($model) as $k => $v)
  99                  $this->$k = $v;
 100              // append validation rules
 101              $this->val_rules = array_merge($this->val_rules, $model->rules());
 102              #Yii::log('merged rules : '.print_r($this->val_rules,true), 'debug', 'DynamicModelBase::addMixinModel()');
 103              // append labels
 104              $this->attr_labels = array_merge($this->attr_labels, $model->attributeLabels());
 105              #Yii::log('merged labels : '.print_r($this->attr_labels,true), 'debug', 'DynamicModelBase::addMixinModel()');
 106              }
 107          else
 108              throw new CException(Yii::t('yii','DynamicModelBase::addMixinModel() : param not an instance of FormModelBase', array()));
 109          }

 155  	protected function addValRule($varname,$r='safe')
 156          {
 157          $found = false;
 158          if($this->val_rules==null or !is_array($this->val_rules))
 159              //$this->__construct();
 160              throw new CException(Yii::t('yii','DynamicModelBase: Property "{class}.{property}" (set in ctor) is null or not an array.', array('{class}'=>get_class($this), '{property}'=>'val_rules')));
 161  
 162          foreach($this->val_rules as &$ar1)
 163              {
 164              if($ar1[1] == $r)
 165                  {
 166                  $found = true;
 167                  $rs = $ar1[0];
 168                  $ar1[0] = $rs.','.$varname;
 169                  }
 170              }
 171          if(!$found)
 172              {
 173              $this->val_rules[] = array($varname,$r);
 174              }
 175          }

 177      /**
 178       * attribute accessor
 179       * 
 180       * Overridden to NOT throw an exception, as the base implementation does.
 181       * Also, wrapping an invocation of the parent function in try/catch also
 182       * mysteriously failed to work, presumably because of bypassing by PHP 5.3
 183       * implementation of the magic __set/__get functions
 184       * 
 185       * @param  string $name attribute name
 186       * @return mixed  attribute value
 187       * @access public
 188       */
 189  	public function __get($name)
 190          {
 191          // same as CComponent except ...
 192          $getter='get'.$name;
 193          if(method_exists($this,$getter))
 194              return $this->$getter();
 195          else if(strncasecmp($name,'on',2)===0 && method_exists($this,$name))
 196              {
 197              // duplicating getEventHandlers() here for performance
 198              $name=strtolower($name);
 199              if(!isset($this->_e[$name]))
 200                  $this->_e[$name]=new CList;
 201              return $this->_e[$name];
 202              }
 203          else if(isset($this->_m[$name]))
 204              return $this->_m[$name];
 205          else if(is_array($this->_m))
 206              {
 207              foreach($this->_m as $object)
 208                  {
 209                  if($object->getEnabled() && (property_exists($object,$name) || $object->canGetProperty($name)))
 210                      return $object->$name;
 211                  }
 212              }
 213          // .. CComponent throws exception here - we just get the darn value
 214          return $object->$name;
 215          }

 220       * Overridden to NOT throw an exception, as the base implementation does.

 231          {
 232          // same as CComponent except ...
 233          $setter='set'.$name;
 234          if(method_exists($this,$setter))
 235              return $this->$setter($value);
 236          else if(strncasecmp($name,'on',2)===0 && method_exists($this,$name))
 237              {
 238              // duplicating getEventHandlers() here for performance
 239              $name=strtolower($name);

 242              return $this->_e[$name]->add($value);
 243              }
 244          else if(is_array($this->_m))
 245              {
 246              foreach($this->_m as $object)
 247                  {
 248                  if($object->getEnabled() && (property_exists($object,$name) || $object->canSetProperty($name)))
 249                      return $object->$name=$value;
 250                  }
 251              }
 252          // .. CComponent throws exception here - we just set the darn value
 253          return $this->$name=$value;
 254          }
