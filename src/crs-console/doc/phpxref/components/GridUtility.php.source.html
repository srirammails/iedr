  80      /**
  81       * returns array of year-month values for select-lists
  82       * 
  83       * @return array     array of 10 strings like '2011-06' => 'June-2011', for next 10 months
  84       * @access protected
  85       * @static
  86       */
  87  	static protected function getMnYrSelOpts()
  88          {
  89          // start at previous calendar month
  90          $t = strtotime(date('Y-m-01'));
  91          $selarr = array();
  92          for($i=0; $i<10; $i++)
  93              {
  94              $tm = strtotime('+'.$i.' month',$t);
  95              $selarr[date('Y-m',$tm)] = date('F-Y',$tm);
  96              }
  97          return $selarr;
  98          }

 100      /**
 101       * returns a two-element array of last year and this year as integers
 102       * 
 103       * @return array     two-element array of last year and this year as integers
 104       * @access protected
 105       * @static
 106       */
 107  	static protected function getYearSelOpts()
 108          {
 109          // start at previous year
 110          $selarr = array();
 111          $Y = date('Y',strtotime('-1 year'));
 112          $selarr[$Y] = $Y;
 113          $Y = date('Y');
 114          $selarr[$Y] = $Y;
 115          return $selarr;
 116          }

 118      /**
 119       * returns an array of months for select-lists
 120       * 
 121       * @return array  array of months, key and value are both month-name strings
 122       * @access public
 123       * @static
 124       */
 125  	static public function getMonthSelOpts()
 126          {
 127          // start at Jan
 128          $arr = array();
 129          foreach(self::getMonthList() as $i => $m)
 130              $arr[$m] = $m;
 131          return $arr;
 132          }

 134      /**

 136       * 
 137       * @return array  array of (int,1-12)month => (string,eg 'January')month-name
 138       * @access public
 139       * @static

 141  	static public function getMonthList()
 142          {
 143          $l = array();
 144          for($i=1;$i<13;$i++)
 145              $l[$i] = date('F',strtotime('2000-'.$i.'-01'));
 146          return $l;
 147          }
