  42  	protected function validateAttribute($model,$attribute)
  43          {

  46              $value=$model->$attribute;

  87      /**
  88       * Gets an account ID for a given NicHandle, returning a cached value of found, otherwise calling {@link _NICExists}
  89       * 
  90       * @param  string    $nic Nic Handle to search for
  91       * @return integer   Account ID, if found, otherwise -1
  92       * @access protected
  93       */
  94      protected function NICExists($nic)
  95          {
  96          $accno = -1;
  97          // is cached ?
  98          $c = Yii::app()->cache;
  99          // cache is global to application (all session/users), so use a cache key that contains the logged-in user,
 100          // to avoid returning a cached value to another user
 101          $ckey = 'NICExistsFor_'.Yii::app()->user->authenticatedUser->username.'_'.$nic;
 102          if($c!=null)
 103              {
 104              $cached_accno = null;
 105              $cached_accno = Yii::app()->cache->get($ckey);
 106              if(is_integer($cached_accno))
 107                  {
 108                  Yii::log('cache hit for '.$ckey, 'debug', 'NicHandleValidator::NICExists('.$nic.')');
 109                  $accno = $cached_accno;
 110                  }
 111              else
 112                  {
 113                  Yii::log('cache miss for '.$ckey, 'debug', 'NicHandleValidator::NICExists('.$nic.')');
 114                  $accno = $this->_NICExists($nic);
 115                  Yii::app()->cache->set($ckey, $accno, 60);
 116                  }
 117              }
 118          else
 119              {
 120              Yii::log('no caching', 'debug', 'NicHandleValidator::NICExists('.$nic.')');
 121              $ret = $this->_NICExists($nic);
 122              }
 123          return $accno > 0 ? true : false;
 124          }
