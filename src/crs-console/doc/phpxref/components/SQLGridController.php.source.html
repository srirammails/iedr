 122          {
 123          $pagedata = array('records'=>0,'page'=>$pagenum,'total'=>0,'rows'=>array());

 130          // get data
 131          $connection = Yii::app()->db;
 132          $command = $connection->createCommand($query);
 133          $result = $command->query();
 134          if($totalRsts > 0)
 135                  foreach($result as $k => $o) {
 136                  Yii::log('fetched line $k => $o '.$k.' => '.print_r($o,true), 'debug', 'SQLGridController::getData');
 137                  $pagedata['rows'][] = $model->addResults($o);
 138                  }
 139          return $pagedata;

 149  	public function criteriaToSQL($criteria)
 150          {
 151          $sqlcrit = '';
 152          foreach($criteria as $fld => $val)
 153              {
 154              if($fld!=null and $val!=null and $val!='')
 155              $sqlcrit .= 'AND '.$fld.' '.(strpos($val,'%')>0?'LIKE': '=').'\''.$val.'\' ';
 156              }
 157          return $sqlcrit;
 158          }

 160      /**
 161       * returns the SQL query to be used for retrieving a page of data for a jqGrid
 162       * 
 163       * @param  CModel  $model    model, passed to $this->getPartialQuery()
 164       * @param  array   $criteria array of constraints in the form of (column-name => data-value)
 165       * @param  object  $sort     Sorting data object, withattributes 'sortBy' (SQL column name) and boolean 'ascending'.
 166       * @param  integer $perpage  how many rows on each page 
 167       * @param  integer $pagenum  which page of paginated results to fetch; 1-based
 168       * @return string  SQL query to be used for retrieving a page of data for a jqGrid
 169       * @access public 
 170       */
 171  	public function getQuery($model,$criteria,$sort,$perpage,$pagenum)
 172          {
 173          $offset = $perpage * ($pagenum-1);
 174          $limit  = $perpage;
 175          $qryParams = $sort ? ('ORDER BY '.$sort->sortBy . (($sort->ascending) ? ' ASC':' DESC')) : '';
 176          $partialQry = $this->getPartialQuery($model,$criteria);
 177          return $partialQry.' '.$qryParams.' LIMIT '.$offset.','.($offset+$limit).';';
 178          }
