  21  class UserIdentity extends CUserIdentity
  22  {
  23      /**
  24       * Authenticates a user.
  25       * The example implementation makes sure if the username and password

  27       * In practical applications, this should be changed to authenticate
  28       * against some persistent user identity storage (e.g. database).
  29       * @return boolean whether authentication succeeds.
  30       */
  31  	public function authenticate()
  32      {
  33          $users=array(
  34              // username => password
  35              'demo'=>'demo',
  36              'admin'=>'admin',
  37          );
  38          if(!isset($users[$this->username]))
  39              $this->errorCode=self::ERROR_USERNAME_INVALID;
  40          else if($users[$this->username]!==$this->password)
  41              $this->errorCode=self::ERROR_PASSWORD_INVALID;
  42          else
  43              $this->errorCode=self::ERROR_NONE;
  44          return !$this->errorCode;
  45      }
  46  }
