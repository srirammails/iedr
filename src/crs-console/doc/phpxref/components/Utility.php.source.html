 179      /**
 180       * split a delimited string into an array, then validate the strings with {@link IEDomainValidator}
 181       * 
 182       * @param  unknown $in       string containing one or more domains delimited by newline, comma, colon, semicolon, or space
 183       * @param  array   &$errDoms array will have added to it domains from the input which are not valid .IE domains
 184       * @return array   array of valid .IE domain names
 185       * @access public 
 186       * @static
 187       */
 188          static public function parseInput($in,&$errDoms=null)
 189          {
 190          // parse input string into array and return an array of valid domain-names
 191          if($errDoms==null) $errDoms = array();
 192          $validIEDomains = array();
 193          $tokens = preg_split('/[\s,;:]+/', strtolower($in));
 194          #Yii::log('str "'.$in.'" into array: '.print_r($tokens,true), 'debug', 'Utility::parseInput()1');
 195          foreach($tokens as $d)
 196              if(preg_match(IEDomainValidator::$ie_domain_regex,$d)==1)
 197                  $validIEDomains[$d] = 1;
 198              else
 199                  $errDoms[$d] = 1;
 200          $validIEDomains = array_keys($validIEDomains);
 201          #Yii::log('str "'.$in.'": valid = '.print_r($validIEDomains,true), 'debug', 'Utility::parseInput()2');
 202          $errDoms = array_keys($errDoms);
 203          #Yii::log('str "'.$in.'": errors = '.print_r($errDoms,true), 'debug', 'Utility::parseInput()3');
 204          return $validIEDomains;
 205          }

 236      /**
 237       * returns boolean whether current request matches rules for determining of the access is from an internal network
 238       * 
 239       * @param  unknown $ipAddr IP address
 240       * @return boolean whether current request matches rules for determining of the access is from an internal network
 241       * @access public 
 242       * @static
 243       */
 244  	static public function isInternalNetwork($ipAddr=null)
 245          {
 246          if($ipAddr==null)
 247              $ipAddr = $_SERVER['REMOTE_ADDR'];
 248          $internalNet = Yii::app()->params['internal_network'];
 249          foreach($internalNet['exact_matches'] as $exactIp)
 250              if($ipAddr == $exactIp)
 251                  return true;
 252          foreach($internalNet['regex_matches'] as $IpRegex)
 253              if(preg_match($IpRegex, $ipAddr)==1)
 254                  return true;
 255          return false;
 256          }
