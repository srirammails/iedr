 103      /**
 104       * Formats CSV data as a string, then returns it.
 105       * 
 106       * Formats CSV data as a string, then returns it, using a temporary file (see here: http://php.net/manual/en/wrappers.php.php)
 107       * 
 108       * @return string CSV data
 109       * @access public
 110       */
 111  	public function output() {
 112          $csvdata = '';
 113          // open temp output file
 114          $out = fopen('php://temp', 'rwb');
 115          // output column headers
 116          $n = 0;
 117          foreach($this->colheads as $col) {
 118              $l = $this->strdelim.$col.$this->strdelim;
 119              if(++$n!=1) fputs($out, $this->fldsep);
 120              fputs($out, $l);
 121              }
 122          fputs($out, "\n"); // unix newline here, convert later
 123          // output data
 124          $buffer = '';
 125          foreach($this->data as $row)
 126              fputcsv($out, $row, $this->fldsep, $this->strdelim);
 127          // re-read output, replacing unix-newline with dos newline
 128          fseek($out, 0);
 129          while (($buffer = fgets($out, 4096)) !== false)
 130              $csvdata .= str_replace("\n","\r\n",$buffer);
 131          fclose($out);
 132  
 133          return $csvdata;
 134          }
 135      }
