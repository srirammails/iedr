 101      /**
 102       * return an array of key-value pairs in a jqGrid colmodel select-list format string
 103       * 
 104       * see {@link http://www.trirand.com/jqgridwiki/doku.php?id=wiki:search_config} table 'property', row 'value'

 106       * @param  array  $arr array of key-value pairs for a select-list based search

 108       * @access public
 109       */
 110  	public function arrayAsColModelSelectOptions($arr)
 111          {
 112          $s = ':(any);'; $i=0; $n=count($arr);
 113          foreach($arr as $k => $v)
 114              $s .= $k.':'.$v.(++$i < $n ? ';' : '');
 115          return $s;

 134      /**
 135       * Add the current request's search-filter parameters to the search-object-instance returned by {@link getSearchBase}
 136       * 
 137       * The request filter parameters are sent by the jqGrid as a JSON-encoded structure, referencing colModel column names and values.
 138       * This function is called after JSON-demarshalling (as in 'C' below) and must determine which criteria field to use based on the
 139       * 'field' filter field (by getting the 'criteriafield' member in the {@link columns} array), and add the filter value to the criteria object instance,
 140       * as show in step 'D' below.
 141       * 
 142       * (A) HTTP request
 143       * 
 144       *     GET ...&filters=%7B%22groupOp%22%3A%22AND%22%2C%22rules%22%3A%5B%7B%22field%22%3A%22A%22%2C%22op%22%3A%22bw%22%2C%22data%22%3A%22blahblah%22%7D%5D%7D
 145       * 
 146       * (B) URL-decoded filter
 147       * 
 148       *     [filters] => {"groupOp":"AND","rules":[{"field":"A","op":"bw","data":"blahblah"}]}
 149       * 
 150       * (C) JSON-Decoded filter rules
 151       *    (note: jqGrid filter-operation 'bw' is 'begins-with', though in NRC we interpret all filters as 'contains')
 152       * 
 153       *         [field] => A
 154       *         [op] => bw
 155       *         [data] => blahblah
 156       * 
 157       * (D) search-criteria object filled with filter value(s)
 158       * 
 159       *     CRSDomainAppService_domainSearchCriteria Object
 160       *         (
 161       *         ...
 162       *         [domainName] => blahblah
 163       *         ...
 164       *         )
 165       * 
 166       * 
 167       * @param  array  $searchparams deserialised jqGrid search filter
 168       * @return object Return WSAPI search-criteria object instance
 169       * @access public
 170       */
 171  	public function getSearch($searchparams)
 172          {
 173          Yii::log('searchparams = '.print_r($searchparams,true), 'debug', 'GridModelBase::getSearch()');
 174          $criteria = $this->getSearchBase();
 175          foreach($searchparams as $k => $r)
 176              {
 177              Yii::log('searchparam k='.print_r($k,true).' => '.print_r($r,true), 'debug', 'GridModelBase::getSearch()');
 178              $f = $this->columns[$r->field]['criteriafield'];
 179              Yii::log('criteriafield='.print_r($f,true), 'debug', 'GridModelBase::getSearch()');
 180              if($f != null)
 181                  {
 182                  $d = $r->data;
 183                  if($r->wildcardpadding)
 184                      {
 185                      switch($r->wildcardpadding)
 186                          {
 187                          case 'NONE' : break;
 188                          case 'LEFT' : $d = '%'.$d   ;  break;
 189                          case 'RIGHT': $d =     $d.'%'; break;
 190                          case 'BOTH' : default: $d = '%'.$d.'%'; break;
 191                          }
 192                      }
 193                  $criteria->$f = $d;
 194                  }
 195              }
 196          Yii::log('returning '.print_r($criteria,true), 'debug', 'GridModelBase::getSearch()');
 197          return $criteria;
 198          }
