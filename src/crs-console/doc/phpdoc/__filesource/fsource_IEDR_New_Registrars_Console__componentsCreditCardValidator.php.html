  67.     /**
  68.      * Credit Card Number validation regex patterns
  69.      * 
  70.      * - Visa: All numbers start with a 4. New cards have 16 digits. Old cards have 13.
  71.      * 
  72.      * - MasterCard: All numbers start with the numbers 51 through 55. All have 16 digits.
  73.      * 
  74.      * - Laser : They should have between 16 and 19 numbers inclusively, and should have a prefix of one of 6304, 6706, 6709
  75.      * 
  76.      * @var    array 
  77.      * @access public

  80.      */
  81.     static $card_patterns = array(
  82.         # Visa: All numbers start with a 4. New cards have 16 digits. Old cards have 13.
  83.         CREDCARDTYPE_VISA   => '4[0-9]{12}(?:[0-9]{3})?',
  84.         # MasterCard: All numbers start with the numbers 51 through 55. All have 16 digits.
  85.         CREDCARDTYPE_MASTERCARD     => '5[1-5][0-9]{14}',
  86.         # switch
  87.         #'SWITCH' => '([0-9]{6}[0-9]{12}|[0-9]{18})',
  88.         # American Express: numbers start with 34 or 37 and have 15 digits.
  89.         #'AMEX '  => '3[47][0-9]{13}',
  90.         # laser : They should have between 16 and 19 numbers inclusively, and should have a prefix of one of 6304, 6706, 6709
  91.         CREDCARDTYPE_LASER  => '6(304|706|709)([0-9]{12,15})',
  92.         # Diners Club: numbers begin with 300 through 305, 36 or 38. All have 14 digits.
  93.         #     There are cards that begin with 5 and have 16 digits;
  94.         #     These are a joint venture between Diners Club and MasterCard, and should be processed like a MasterCard.
  95.         #'DINERS' => '3(?:0[0-5]|[68][0-9])[0-9]{11}',
  96.         );

  130.     /**

  179.     /**
  180.      * perform Luhn check
  181.      * 
  182.      * inspired by http://www.codepost.org/browse/snippets/88
  183.      * 
  184.      * @param  string $str credit card number
  185.      * @return boolean check results
  186.      * @access public
  187.      * @static
  188.      * @see http://en.wikipedia.org/wiki/Luhn_algorithm
  189.      */
  190.     public static function CCLuhnCheck($str)
  191.         {
  192.         $odd = true;
  193.         $sum = 0;
  194.         foreach( array_reverse(str_split($str)) as $num)
  195.             $sum += array_sum( str_split(($odd = !$odd) ? $num*2 : $num) );
  196.         return (($sum % 10 == 0) && ($sum != 0));
  197.         }

  201.      *

  203.      * 
  204.      * @param  string $num  credit card number
  205.      * @param  string &$typ optional type ; if null, will be set to the type of the first regex the card number was found to match, if any
  206.      * @return boolean whether card matched regex for the specified type, or if type was null, if card matched regex of any type
  207.      * @access public
  208.      * @static

  210.     public static function CCPrefixLengthTypeCheck($num,&$typ=null)
  211.         {
  212.         $pass = false;
  213.         if($typ!=null and isset(CreditCardValidator::$card_patterns[$typ]))
  214.             $pass = preg_match('/^'.CreditCardValidator::$card_patterns[$typ].'$/',$num);
  215.         else
  216.             foreach(CreditCardValidator::$card_patterns as $cardtype => $regex)
  217.                 if(CreditCardValidator::CCPrefixLengthTypeCheck($num,$cardtype))
  218.                     {
  219.                     $pass = true;
  220.                     $typ  = $cardtype;
  221.                     break;
  222.                     }
  223.         return $pass;
  224.         }
