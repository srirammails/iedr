  241.      * returns partial query sql string for data retrieval with parameters
  242.      *

  244.      * @param  string  $criteria search criteria
  245.      * @return string  sql string
  246.      * @access public
  247.      */
  248.     public function getPartialQuery($model,$criteria)
  249.         {
  250.         $u = Yii::app()->user->name;
  251.         $dt = '';
  252.         switch($this->type)
  253.             {
  254.             case 'nrc_cn_on_invs_msddels':
  255.                 $dt = $this->getDateStr(array('I_Inv_Dt'),$model); break;
  256.             case 'nrc_cn_on_invs_msd_xfer': case 'nrc_cn_on_invs_xfer_away_pp':
  257.                  $dt = $this->getDateStr(array('C_Nt_Ts'),$model); break;
  258.             }
  259.         $query = '';
  260.         switch($this->type)
  261.             {
  262.             case 'nrc_cn_deletions':
  263.                 $query = "SELECT Domain, Holder, InvoiceNo, InvoiceDate, RenewalDate, RegistrationDate, Amount, I_Inv_Vat_Amt FROM nrc_cn_deletions WHERE I_Bill_NH ='".$u."'";
  264.                 break;
  265.             case 'nrc_cn_on_invs_msddels':
  266.                 $query = "SELECT I_DName, I_Inv_Dt, I_Inv_No, Cr_Note_Dt FROM nrc_cn_on_invs_msddels WHERE I_Bill_Nh ='".$u."'  AND ".$dt;
  267.                 break;
  268.             case 'nrc_cn_on_invs_msd_xfer':
  269.                 $query = "SELECT I_DName, I_Inv_Dt, I_Inv_No, Tr_Date, Type, C_Nt_Ts FROM nrc_cn_on_invs_msd_xfer WHERE Old_Nic_Handle ='".$u."'  AND ".$dt;
  270.                 break;
  271.             case 'nrc_cn_on_invs_xfer_away_pp':
  272.                 $query = "SELECT I_DName, I_Inv_Dt, I_Inv_No, Tr_Date, Type, C_Nt_Ts FROM nrc_cn_on_invs_xfer_away_pp WHERE I_Bill_NH ='".$u."' AND ".$dt;

  274.             case 'nrc_cn_xfer_away':
  275.                 $query = "SELECT Domain, Holder, InvoiceNo, InvoiceDate, RenewalDate, RegistrationDate, Amount, I_Inv_Vat_Amt FROM nrc_cn_xfer_away WHERE I_Bill_NH ='".$u."'";
