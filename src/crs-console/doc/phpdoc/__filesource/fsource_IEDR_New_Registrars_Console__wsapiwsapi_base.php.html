  132. /**
  133.  * Recursively adds and object or array to a given object's data member
  134.  * 
  135.  * When data in the form of nested arrays or objects must be added to
  136.  * an model which has the same data model, but is flattened, then this function
  137.  * is called to set all the model data mambers to the values in the nested source data.
  138.  * As Yii data Models cannot contain nested arrays or objects, they must represent
  139.  * tree structures in a flattened format; for example a Soap data structure like this
  140.  * <ul>
  141.  *     <li> name
  142.  *     <li> address
  143.  *         <ul>
  144.  *             <li> street
  145.  *             <li> city
  146.  *         </ul>
  147.  * </ul>
  148.  * would be represented in the model like this:
  149.  * <ul>
  150.  *     <li> public $stuff_name;
  151.  *     <li> public $stuff_address_street;
  152.  *     <li> public $stuff_address_city;
  153.  * </ul>
  154.  * So to set the model from the object, one would call this function like this:
  155.  * <ul>
  156.  *     <li> addToObjectAsProperties($model, 'stuff', $SoapData);
  157.  * </ul>
  158.  *
  159.  * @param  object $o object (e.g. a model) to add data to
  160.  * @param  string $n object public data member name
  161.  * @param  array  $a object or array to add (e.g. an object returned by a soap call)
  162.  * @param  string $s separator used to construct target object data membernames
  163.  * @return void 
  164.  */
  165. function addToObjectAsProperties($o,$n,$a,$s='_')
  166.     {
  167.     // NOTE this doesn't work for Yii objects which implement __set() !
  168.     if(!is_object($o)) die("addToObjectAsProperties: first param (object to argument) must be an object");
  169.     if(!is_string($n)) die("addToObjectAsProperties: second param (name of member) must be a string");
  170.     if(!is_string($s)) die("addToObjectAsProperties: optional fourth param (separator) must be a string");
  171.  
  172.     if(is_object($a))
  173.         addToObjectAsProperties($o,$n,objectToArray($a),$s);
  174.     else
  175.         if(is_array($a))
  176.             foreach($a as $k => $v)
  177.                 addToObjectAsProperties($o, $n.$s.$k, $v, $s);
  178.         else
  179.             $o->$n = $a;
  180.     }
