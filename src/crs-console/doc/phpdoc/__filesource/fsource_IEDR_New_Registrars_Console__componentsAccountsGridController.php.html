  101.         {
  102.         // called by GridController::actionConfirmaction()

  114.         if($gsc_model->command == 'msd')

  119.             {

  138.                         {
  139.                         $totalFee += $result->amount;
  140.                         $totalVat += $result->vat;
  141.                         foreach($this->groupCommandByPeriod($gsc_model) as $period => $domainlist)
  142.                             {
  143.                             // the results are for the whole collection of periods and domains, but we display results PER-Period
  144.                             $result->amount = null;
  145.                             $result->vat    = null;
  146.                             $gsc_model->commandresults[$period] = array('domainlist'=>$domainlist,'results'=>$result,'errs'=>null);
  147.                             }
  148.                         }

  151.                     break;
  152.                 default:
  153.                     foreach($this->groupCommandByPeriod($gsc_model) as $period => $domainlist)
  154.                         {
  155.                         $result = null;
  156.                         switch($gsc_model->command)

  160.                                 break;
  161.                             case 'off':

  166.                                 CRSPaymentAppService_service::payMsdReactivation($result,$errs,Yii::app()->user->authenticatedUser,$pRequest,$remoteAddr,$domainlist,$gsc_model->needCreditCard,$isTest);
  167.                                 break;
  168.                             }
  169.                         if($result!=null and is_object($result))
  170.                             {
  171.                             $totalFee += $result->amount;
  172.                             $totalVat += $result->vat;
  173.                             }
  174.                         else
  175.                             $this->backend_errors[$period] = $errs;

  179.                 }
  180.  
  181.             }

  289.     public function getInvoicesSummary($invoiceType=null,$month=null,$domainName=null)
  290.         {
  291.         if($invoiceType==null) $invoiceType = $this->getNeededInvoiceType();
  292.         $result = null;
  293.         /*
