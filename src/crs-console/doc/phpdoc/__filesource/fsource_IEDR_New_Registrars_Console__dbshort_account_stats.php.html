  21. function microtime_float()

  31.  * @return mixed  column 'x' of first row of results
  32.  */
  33. function run_stats_qry($sql)
  34.     {
  35.     $d=DB::get_db();
  36.     $d->query($sql,__FILE__,__LINE__);
  37.     if($r=$d->fetch())
  38.         return $r->x;

  40.     }

  49. function get_cached_stats_query($cache_key,$sql)
  50.     {

  53.     if($s==null)
  54.         {
  55.         $tstart = microtime_float();
  56.         $s = run_stats_qry($sql);
  57.         $tend = microtime_float();
  58.         $elapsed = $tend - $tstart;
  59.         $timeout = 600; // 10 min
  60.         Yii::app()->cache->set($cache_key, $s, $timeout);
  61.         #Yii::log('query for '.$cache_key.' took '.$elapsed.' seconds', 'debug', 'short_account_stats::get_cached_stats_query()'.__LINE__);
  62.         }
  63.     return $s;
  64.     }
