  155.         {
  156.         // call with a parameter being either an instance of this class or a class that has it's values mixed in

  159.         $payreq->currency       = 'EUR';
  160.         $payreq->amount        = $mdl->euros_amount * 100;
  161.         $payreq->cardNumber     = $mdl->creditcardno;
  162.         $payreq->cardExpDate    = sprintf('%02u%02u',$mdl->exp_month,$mdl->exp_year);
  163.         $payreq->cardHolderName    = $mdl->holder;
  164.  
  165.         // determine the card type
  166.         CreditCardValidator::CCPrefixLengthTypeCheck($mdl->creditcardno,$payreq->cardType);
  167.         switch($payreq->cardType)
  168.             {
  169.             case CREDCARDTYPE_VISA:
  170.             case CREDCARDTYPE_MASTERCARD:
  171.                 $payreq->cvnPresenceIndicator = CVN_PRESENT_OK;
  172.                 break;
  173.             case CREDCARDTYPE_LASER:
  174.                 $payreq->cvnPresenceIndicator = CVN_PRESENT_NOT_ON_CARD;
  175.                 break;
  176.             }
  177.         $payreq->cvnNumber = $mdl->cvn;
  178.  
  179.         return $payreq;
