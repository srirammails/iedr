  84.     /**
  85.      * Deserialises an object or array
  86.      * 
  87.      * Deserialises an object or array, detecting (with mini header strings like 'W?W=')
  88.      * different decoding steps:
  89.      * <ul>
  90.      * <li> 'WUW' : urldecode
  91.      * <li> 'WBW' : base64
  92.      * <li> 'WZW' : gzuncompress (not used, as the base64 decode does not reliably produce the same number of characters every time ;
  93.      * <li> 'WKW' : ({@link Yii::app()->getSecurityManager()->validateData http://www.yiiframework.com/doc/api/1.1/CSecurityManager#validateData-detail})
  94.      * <li> 'WSW' : unserialize
  95.      * </ul>
  96.  
  97.      * 
  98.      * @param  string $str output of @safeSerializeObj
  99.      * @return object an instantiated object instance, or null
  100.      * @access public
  101.      * @static
  102.      */
  103.     public static function safeDeserializeObj($str)
  104.         {
  105.         // whether the data arrives url-encoded or not is unpredictable : test and proceed with caution ...
  106.         // test : php -r "print_r(unserialize(substr(gzuncompress(substr(base64_decode(substr(urldecode(substr('$STR',4)),4)),4)),48)));" ; echo
  107.         $obj = null;
  108.         $data = $str;
  109.         #self::log('INSTR='.$data, 'debug', 'AuthOnlyBaseController::safeDeserializeObj()');
  110.         while(strpos($data,'WUW%')===0) {
  111.             $data=urldecode($data);
  112.             #self::log('URLDECOD1={'.strlen($data).'}='.$data, 'debug', 'AuthOnlyBaseController::safeDeserializeObj()');
  113.             }
  114.         if(strpos($data,'WUW=')===0) {
  115.             $data=urldecode(substr($data,4));
  116.             #self::log('URLDECOD2={'.strlen($data).'}='.$data, 'debug', 'AuthOnlyBaseController::safeDeserializeObj()');
  117.             }
  118.         if(strpos($data,'WBW=')===0) {
  119.             $data=base64_decode(substr($data,4));
  120.             #self::log('64DECOD(UU)={'.strlen($data).'}='.urlencode($data), 'debug', 'AuthOnlyBaseController::safeDeserializeObj()');
  121.             }
  122.         if(extension_loaded('zlib') and strpos($data,'WZW=')===0) {
  123.             $data=gzuncompress(substr($data,4));
  124.             #self::log('UNZIP={'.strlen($data).'}='.$data, 'debug', 'AuthOnlyBaseController::safeDeserializeObj()');
  125.             }
  126.         if(strpos($data,'WKW=')===0) {
  127.             $data=Yii::app()->getSecurityManager()->validateData(substr($data,4));
  128.             #self::log('UNHASH={'.strlen($data).'}='.$data, 'debug', 'AuthOnlyBaseController::safeDeserializeObj()');
  129.             }
  130.         if(strpos($data,'WSW=')===0) {
  131.             $obj = unserialize(substr($data,4));
  132.             #self::log('DESER={'.strlen($data).'}='.print_r($obj,true), 'debug', 'AuthOnlyBaseController::safeDeserializeObj()');
  133.             }
  134.  
  135.         return $obj;
  136.         }
