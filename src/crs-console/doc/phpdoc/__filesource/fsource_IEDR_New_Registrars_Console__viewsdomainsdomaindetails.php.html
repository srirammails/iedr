  66. function getDataAndPostForm(mode,nictype,srcf,serializeUrl,editNIC) {
  67.     // this fn called on click (link or button) does:
  68.     // (1) send form to server, gets serialized model back
  69.     // (2) sets up hidden form in this page
  70.     // (3) submits the form, navigating away to the next page
  71.  
  72.     var dispatchUrl;
  73.     switch(mode) {
  74.         case "new":  dispatchUrl = 'index.php?r=nichandles/createnichandle'; break;
  75.         case "find": dispatchUrl = 'index.php?r=nicsearch/namesearch'; break;
  76.         case "edit":
  77.             var theNic = $('#'+editNIC).val();
  78.             dispatchUrl = 'index.php?r=nichandles/editnichandle&id='+theNic;
  79.             $("#hid #nic").val(theNic);
  80.             break;
  81.         }
  82.     // (1) post form ('srcf') to server, retrieve serialized model via ajax
  83.     $.post(serializeUrl, $("#"+srcf).serialize(), function(ajaxdata) {
  84.         // (2) set up form
  85.         $("#hid").attr("action",dispatchUrl);
  86.         $("#hid #data").val(ajaxdata); // the serialised form data
  87.         $("#hid #mode").val(mode); // new/find/edit
  88.         $("#hid #nictype").val(nictype); // which nic role for the mode (for updating the correct field, on return to this page)
  89.         // Yii doesn't set an ID for the token hidden-input, so find by name
  90.         var itok = $('#'+srcf+' input[name="YII_CSRF_TOKEN"]')[0];
  91.         // set hidform token value
  92.         var hitok = $("#hid #tok")[0];
  93.         $(hitok).val($(itok).val());
  94.         // (3) post form with serialized model
  95.         $("#hid").submit();
  96.         });
  97.     }
