  203. /**
  204.  * for the given menu label, locate the label in the menu tree,
  205.  * then print a row of '/'-separated links for each parent-or-self menu item,
  206.  * each one being a hyperlink to the relevant menu-page
  207.  * 
  208.  * @param  string $item label of menu item to print breadcrumb trail for
  209.  * @return void 
  210.  */
  211. function printBreadcrumbTrail($item)
  212.     {
  213.     $m = getMenu();
  214.     checkLabelsAreUnique($m);
  215.     $ma = array('items'=>$m);
  216.     // 1. find the item in the nesting arrays
  217.     $treepath = menuFindLabelRecurse($ma,$item);
  218.     if($treepath==null) throw new Exception('The Menu label "'.$item.'" was NOT FOUND : either change the search-string, or update the menu\'s label (case sensitive)');
  219.     // 2. convert to array
  220.     $tpa = explode(',',$treepath);
  221.     // 3. iterate over found nodes and print text+link
  222.     $dc = new DummyController('');
  223.     foreach($tpa as $n => $l)
  224.         {
  225.         $ma = $ma['items'][$l];
  226.         if($l != 'X') {
  227.             if($n != 0) echo ' / ';
  228.             $tgt = $dc->createUrl($ma['url'][0]);
  229.             echo '<a href="'.$tgt.'">'.$ma['label'].'</a>';
  230.             }
  231.         }
  232.     }

  234. /**
  235.  * Recursively locate a label in a menu tree, returning a comma-separated list of parent item labels
  236.  * 
  237.  * @param  array   $m menu array structure
  238.  * @param  string  $l menu label being sought
  239.  * @param  integer $i recursion depth gauge
  240.  * @return string  comma-separated list of parent item labels
  241.  */
  242. function menuFindLabelRecurse($m,$l,$i=0)
  243.     {
  244.     if($m['label']==$l)
  245.         {
  246.         #echo BR.'found';
  247.         return 'X';
  248.         }
  249.     if(isset($m['items']))
  250.         {
  251.         #echo BR.'iterating..';
  252.         foreach($m['items'] as $idx => $ia)
  253.             {
  254.             #echo BR.'trying in '.$idx;
  255.             $chld = menuFindLabelRecurse($ia,$l,$i+1);
  256.             if($chld!=null)
  257.                 {
  258.                 #echo BR.'found in '.$idx.' in child '.$chld.' at depth '.$i;
  259.                 return $idx.','.$chld;
  260.                 }
  261.             }
  262.         }
  263.     return null;
  264.     }

  266. /**
  267.  * Recursively print a html tree for a menu item; itself and all it's children, each label a hyperlink to the relevant URL
  268.  * 
  269.  * @param  array   $m   menu structure
  270.  * @param  unknown $sel label to start from, or null for all
  271.  * @param  integer $lvl recursion depth gauge
  272.  * @return void 
  273.  */
  274. function printMenuHeirarchy($m,$sel=null,$lvl=0)
  275.     {
  276.     if(is_array($m))
  277.         {
  278.         $sp = str_repeat('  ',$lvl);
  279.         print($sp.'<ul>'.BR);
  280.         $c = new DummyController('');
  281.         foreach($m as $k => $v)
  282.             {
  283.             $u = $c->createUrl($v['url'][0]);
  284.             $l = $v['label'];
  285.             if( ($sel==null or $sel==$l) and ( isset($v['visible']) && $v['visible'] ) )
  286.                 {
  287.                 print($sp.'<li><a href="'.$u.'">'.$l.'</a>'.BR);
  288.                 if(isset($v['items']) and $v['visible'])
  289.                     printMenuHeirarchy($v['items'],null,$lvl+1);
  290.                 else echo ' ...';
  291.                 print($sp.'</li>'.BR);
  292.                 }
  293.             else
  294.                 {
  295.                 if(isset($v['items']) and $v['visible'])
  296.                     printMenuHeirarchy($v['items'],$sel,$lvl+1);
  297.                 }
  298.             }
  299.         print($sp.'</ul>'.BR);
  300.         }
  301.     }

  303. /**
  304.  * Recursively counts menu labels, used in checking for duplicate labels
  305.  * 
  306.  * @param  array   $arr array to append labels and counts to
  307.  * @param  array   $m   menu array structure
  308.  * @return void 
  309.  */
  310. function labelFinder(&$arr ,$m)
  311.     {
  312.     foreach($m as $k => $v)
  313.         {
  314.         $arr[$v['label']] += 1;
  315.         if(isset($v['items']))
  316.             labelFinder($arr,$v['items']);
  317.         }
  318.     }

  320. /**
  321.  * check for duplicate menu labels
  322.  * 
  323.  * @param  array   $m   menu array structure
  324.  * @return void 
  325.  */
  326. function checkLabelsAreUnique($m)
  327.     {
  328.     $a = array();
  329.     labelFinder($a, $m);
  330.     $errs = array();
  331.     foreach($a as $k => $v) if($v > 1) $errs[] = $k;
  332.     if(count($errs)>0)
  333.         {
  334.         throw new Exception('ERROR: Menu Labels are Duplicated : "'.implode('","',$errs).'"');
  335.         }
  336.     }
