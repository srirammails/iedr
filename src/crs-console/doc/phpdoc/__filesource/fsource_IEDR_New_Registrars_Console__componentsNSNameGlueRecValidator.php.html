  18.  * This Validator must be referenced by adding options to the reference

  21.  *

  38. class NSNameGlueRecValidator
  39.     extends CValidator
  40.     {
  41.  
  42.     /**
  43.      * name of model attribute containing corresponding nameserver hostname
  44.      *
  45.      * @var    string

  48.     public $ns_name = null;

  53.      * @var    string

  58.     /**
  59.      * Validates the IPv4 Address of a DNS Nameserver, adds error message if not valid
  60.      * 
  61.      * First checks if the model-attribute referred to by {@link ns_name}, i.e. the namesserver, matches
  62.      * the model-attribute referred to by {@link dname} ; if this matches, then the "Glue Record" validation
  63.      * rule applies:
  64.      * 
  65.      *  - When Domain Name is in the Domain of a Nameserver, the Nameserver MUST have a valid IPv4 Address
  66.      * 
  67.      * When the Domain does not match the Nameserver hostname, the "Non-Glue Record" rule applies:
  68.      * 
  69.      *  - When Domain Name is NOT in the Domain of a Nameserver, the Nameserver MUST NOT have an Address
  70.      * 
  71.      * @param  CModel    $model     model with data to validate
  72.      * @param  string    $attribute name of attribute in model to validate, containing either en empty or null string, OR an IPv4 Address
  73.      * @return boolean   true if valid
  74.      * @access public
  75.      */
  76.     public function validateAttribute($model,$attribute)
  77.         {
  78.         $ret = false;
  79.         $msg = 'ERR';
  80.         if($model && $attribute)
  81.             {
  82.             $x = $this->ns_name; $ns_value = $model->$x;
  83.             $x = $this->dname;   $dname    = $model->$x;
  84.             $ipaddr=$model->$attribute;
  85.             switch($this->any_domainnames_match($dname, $ns_value))
  86.                 {
  87.                 case 'NONE':
  88.                     // No Match: Non-Glue-Record validation applies: Nameserver MUST NOT have an Address
  89.                     $msg='MUST be empty';
  90.                     $ret = (($ipaddr==null) || ($ipaddr==''));
  91.                     break;
  92.                 case 'SOME': case 'ALL':
  93.                     // Some or All Match: Glue-Record validation applies: Nameserver MUST have a Valid IPv4 Address
  94.                     $msg='MUST be filled';
  95.                     $ret = ($ipaddr!=null) && ($ipaddr!='') && (preg_match(IPV4AddressValidator::getRegex(),$ipaddr)==1);
  96.                     break;
  97.                 }
  98.             }
  99.         if(!$ret) $this->err($model,$attribute,$msg);
  100.         return $ret;
  101.         }
  102.  
  103.     /**
  104.      * returns whether, for the input Nameserver Hostname and Domain Name list, None, Some, or All of the domains match the Nameserver
  105.      * 
  106.      * @param  mixed     $dname    single domain name, or array of domain names, in form "X.Y.ie"
  107.      * @param  string    $ns_value nameserver hostname in form "Y.ie"
  108.      * @return string    'NONE', 'SOME', or 'ALL'
  109.      * @access protected
  110.      */
  111.     protected function any_domainnames_match($dname,$ns_value)
  112.         {
  113.         $nd=1; $nm=0;
  114.         if(is_array($dname))
  115.             {
  116.             $nd=count($dname);
  117.             foreach($dname as $k => $v)
  118.                 $nm += preg_match('/\.'.$v.'$/i', $ns_value);
  119.             }
  120.         else
  121.             $nm = preg_match('/\.'.$dname.'$/i', $ns_value);
  122.         return ($nm==0) ? 'NONE' : ($nd==$nm ? 'ALL' : 'SOME');
  123.         }
  124.  
  125.     /**
  126.      * adds an error message to a model about an attribute
  127.      * 
  128.      * @param  CModel    $model     the model to add the error message to

  131.      * @return void 
  132.      * @access protected
  133.      */
  134.     protected function err($model,$attribute,$msg)
  135.         {
  136.         $lbls = $model->attributeLabels();
  137.         $lbl = $lbls[$attribute];
  138.         $model->addError($attribute,Yii::t('yii',$lbl.' '.$msg));
  139.         }
