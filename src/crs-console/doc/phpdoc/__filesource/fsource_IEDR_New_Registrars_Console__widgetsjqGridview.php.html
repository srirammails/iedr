  117. function currencyFormatter(num) {
  118.     /* render a number in html with currency symbol, 2 decimal places, thousands separators, and half-even rounding (http://www.diycalculator.com/sp-round.shtml#A5) */
  119.     csym = "&euro;";
  120.     if(num==null) return csym+' 0.00';
  121.     num = num.toString().replace(/\$|\,/g,'');
  122.     if(isNaN(num))
  123.         num = "0";
  124.     sign = (num == (num = Math.abs(num)));
  125.     num = Math.floor(num*100+0.50000000001);
  126.     cents = num%100;
  127.     num = Math.floor(num/100).toString();
  128.     if(cents<10)
  129.         cents = "0" + cents;
  130.     for (var i = 0; i < Math.floor((num.length-(1+i))/3); i++)
  131.         num = num.substring(0,num.length-(4*i+3))+','+num.substring(num.length-(4*i+3));
  132.     return (((sign)?'':'-')+csym+' '+num+'.'+cents);
  133.     }
